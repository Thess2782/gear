74/// Definition of the cost schedule and other parameterization for the wasm vm.
75///
76/// Its [`Default`] implementation is the designated way to initialize this type. It uses
77/// the benchmarked information supplied by [`Config::WeightInfo`]. All of its fields are
78/// public and can therefore be modified. For example in order to change some of the limits
79/// and set a custom instruction weight version the following code could be used:
80/// ```rust
81/// use pallet_gear::{Schedule, Limits, InstructionWeights, Config};
82///
83/// fn create_schedule<T: Config>() -> Schedule<T> {
84///     Schedule {
85///         limits: Limits {
86///                 globals: 3,
87///                 locals: 3,
88///                 parameters: 3,
89///                 memory_pages: 16,
90///                 table_size: 3,
91///                 br_table_size: 3,
92///                 .. Default::default()
93///             },
94///         instruction_weights: InstructionWeights {
95///                 version: 5,
96///             .. Default::default()
97///         },
98///             .. Default::default()
99///     }
100/// }
101/// ```
102///
103/// # Note
104///
105/// Please make sure to bump the [`InstructionWeights::version`] whenever substantial
106/// changes are made to its values.
107#[cfg_attr(feature = "std", derive(Serialize, Deserialize))]
108#[cfg_attr(feature = "std", serde(bound(serialize = "", deserialize = "")))]

146/// Describes the upper limits on various metrics.
147///
148/// # Note
149///
150/// The values in this struct should never be decreased. The reason is that decreasing those
151/// values will break existing programs which are above the new limits when a
152/// re-instrumentation is triggered.
153#[cfg_attr(feature = "std", derive(Serialize, Deserialize))]
154#[derive(Clone, Encode, Decode, PartialEq, Eq, RuntimeDebug, TypeInfo)]
155pub struct Limits {
156    /// Maximum allowed stack height in number of elements.
157    ///
158    /// See <https://wiki.parity.io/WebAssembly-StackHeight> to find out
159    /// how the stack frame cost is calculated. Each element can be of one of the
160    /// wasm value types. This means the maximum size per element is 64bit.
161    ///
162    /// # Note
163    ///
164    /// It is safe to disable (pass `None`) the `stack_height` when the execution engine
165    /// is part of the runtime and hence there can be no indeterminism between different
166    /// client resident execution engines.
167    pub stack_height: Option<u32>,
168
169    /// Maximum number of globals a module is allowed to declare.
170    ///
171    /// Globals are not limited through the linear memory limit `memory_pages`.
172    pub globals: u32,
173
174    /// Maximum number of locals a function can have.
175    ///
176    /// As wasm engine initializes each of the local, we need to limit their number to confine
177    /// execution costs.
178    pub locals: u32,
179
180    /// Maximum numbers of parameters a function can have.
181    ///
182    /// Those need to be limited to prevent a potentially exploitable interaction with
183    /// the stack height instrumentation: The costs of executing the stack height
184    /// instrumentation for an indirectly called function scales linearly with the amount
185    /// of parameters of this function. Because the stack height instrumentation itself is
186    /// is not weight metered its costs must be static (via this limit) and included in
187    /// the costs of the instructions that cause them (call, call_indirect).
188    pub parameters: u32,
189
190    /// Maximum number of memory pages allowed for a program.
191    pub memory_pages: u16,
192
193    /// Maximum number of elements allowed in a table.
194    ///
195    /// Currently, the only type of element that is allowed in a table is funcref.
196    pub table_size: u32,
197
198    /// Maximum number of elements that can appear as immediate value to the br_table instruction.
199    pub br_table_size: u32,
200
201    /// The maximum length of a subject in bytes used for PRNG generation.
202    pub subject_len: u32,
203
204    /// The maximum nesting level of the call stack.
205    pub call_depth: u32,
206
207    /// The maximum size of a message payload in bytes.
208    pub payload_len: u32,
209
210    /// The maximum length of a program code in bytes. This limit applies to the instrumented
211    /// version of the code. Therefore `instantiate_with_code` can fail even when supplying
212    /// a wasm binary below this maximum size.
213    pub code_len: u32,
214
215    /// The maximum number of wasm data segments allowed for a program.
216    pub data_segments_amount: u32,
217}

219/// Describes the weight for all categories of supported wasm instructions.
220///
221/// There there is one field for each wasm instruction that describes the weight to
222/// execute one instruction of that name. There are a few exceptions:
223///
224/// 1. If there is a i64 and a i32 variant of an instruction we use the weight
225///    of the former for both.
226/// 2. The following instructions are free of charge because they merely structure the
227///    wasm module and cannot be spammed without making the module invalid (and rejected):
228///    End, Unreachable, Return, Else
229/// 3. The following instructions cannot be benchmarked because they are removed by any
230///    real world execution engine as a preprocessing step and therefore don't yield a
231///    meaningful benchmark result. However, in contrast to the instructions mentioned
232///    in 2. they can be spammed. We price them with the same weight as the "default"
233///    instruction (i64.const): Block, Loop, Nop
234/// 4. We price both i64.const and drop as InstructionWeights.i64const / 2. The reason
235///    for that is that we cannot benchmark either of them on its own but we need their
236///    individual values to derive (by subtraction) the weight of all other instructions
237///    that use them as supporting instructions. Supporting means mainly pushing arguments
238///    and dropping return values in order to maintain a valid module.

570/// Describes the weight for memory interaction.
571///
572/// Each weight with `lazy_pages_` prefix includes weight for storage read,
573/// because for each first page access we need to at least check whether page exists in storage.

575/// This weight is taken in account separately, when loading occurs.
576///
577/// Lazy-pages write accesses does not include cost for uploading page data to storage,
578/// because uploading happens after execution, so benchmarks do not include this cost.
579/// But they include cost for processing changed page data in runtime.

774        .saturating_sub(cost_i64const::<T>().saturating_mul(num_params))
775}

777#[inline]
778fn cost_i64const<T: Config>() -> u32 {
779    type W<T> = <T as Config>::WeightInfo;
780    // Since we cannot directly benchmark the weight of `i64.const` (or `i32.const`; we consider their weights to be the same for our purposes),
781    // we estimate it as the difference between the benchmarks `instr_call_const` and `instr_call`.
782    // The difference between these two benchmarks will give us the weight of `i64.const`, which for x86-64
783    // can be represented by a single `mov` instruction with the embedded const parameter
784    // (for e.x: `mov reg,0xDEADBEAFDEABEAF`).
785    //
786    // This approach may work, but the estimation is not very accurate.
787    // To reduce the impact of this inaccuracy on the assessment of other instructions
788    // (as we subtract the weight of `i64.const` from other instructions),
789    // we introduce a weight division coefficient called `I64CONST_WEIGHT_DIVIDER`.
790    // This helps bring the weight of `i64.const` closer to that of the x86-64 `mov` instruction's estimate.
791    const I64CONST_WEIGHT_DIVIDER: u32 = 2;
792
793    cost_instr_no_params_with_batch_size(W::<T>::instr_i64const) / I64CONST_WEIGHT_DIVIDER
794}

1225impl<T: Config> Default for MemoryWeights<T> {
1226    fn default() -> Self {
1227        // In benchmarks we calculate cost per wasm page,
1228        // so here we must convert it to cost per gear page.
1229        fn to_cost_per_gear_page(w: fn(u32) -> Weight) -> Weight {
1230            Weight::from_parts(
1231                cost(w).ref_time() / (WasmPage::SIZE / GearPage::SIZE) as u64,
1232                0,
1233            )
1234        }
1235
1236        const KB_SIZE: u64 = 1024;
1237
1238        // Memory access thru host function benchmark uses a syscall,
1239        // which accesses memory. So, we have to subtract corresponding syscall weight.
1240        fn host_func_access(w: fn(u32) -> Weight, syscall: fn(u32) -> Weight) -> Weight {
1241            let syscall_per_kb_weight = cost_batched(syscall).ref_time();
1242            let syscall_per_gear_page_weight =
1243                (syscall_per_kb_weight / KB_SIZE) * GearPage::SIZE as u64;
1244
1245            let ref_time = to_cost_per_gear_page(w)
1246                .ref_time()
1247                .saturating_sub(syscall_per_gear_page_weight);
1248
1249            Weight::from_parts(ref_time, 0)
1250        }
1251
1252        const KB_AMOUNT_IN_ONE_GEAR_PAGE: u64 = GearPage::SIZE as u64 / KB_SIZE;
1253        const {
1254            assert!(KB_AMOUNT_IN_ONE_GEAR_PAGE > 0);
1255            assert!(GearPage::SIZE as u64 % KB_SIZE == 0);
1256        }
1257
1258        type W<T> = <T as Config>::WeightInfo;
1259
1260        Self {
1261            lazy_pages_signal_read: to_cost_per_gear_page(W::<T>::lazy_pages_signal_read),
1262            lazy_pages_signal_write: to_cost_per_gear_page(W::<T>::lazy_pages_signal_write),
1263            lazy_pages_signal_write_after_read: to_cost_per_gear_page(
1264                W::<T>::lazy_pages_signal_write_after_read,
1265            ),
1266            lazy_pages_host_func_read: host_func_access(
1267                W::<T>::lazy_pages_host_func_read,
1268                W::<T>::gr_debug_per_kb,
1269            ),
1270            lazy_pages_host_func_write: host_func_access(
1271                W::<T>::lazy_pages_host_func_write,
1272                W::<T>::gr_read_per_kb,
1273            ),
1274            lazy_pages_host_func_write_after_read: host_func_access(
1275                W::<T>::lazy_pages_host_func_write_after_read,
1276                W::<T>::gr_read_per_kb,
1277            ),
1278            // As you can see from calculation: `load_page_data` doesn't include weight for db read.
1279            // This is correct situation, because this weight is already included in above
1280            // lazy-pages weights.
1281            load_page_data: to_cost_per_gear_page(W::<T>::lazy_pages_load_page_storage_data)
1282                .saturating_sub(to_cost_per_gear_page(W::<T>::lazy_pages_signal_read)),
1283            upload_page_data: cost(W::<T>::db_write_per_kb)
1284                .saturating_mul(KB_AMOUNT_IN_ONE_GEAR_PAGE)
1285                .saturating_add(T::DbWeight::get().writes(1)),
1286            mem_grow: cost_batched(W::<T>::mem_grow),
1287            mem_grow_per_page: cost_batched(W::<T>::mem_grow_per_page),
1288            // TODO: make it non-zero for para-chains (issue #2225)
1289            parachain_read_heuristic: Weight::zero(),
1290            _phantom: PhantomData,
1291        }
1292    }
