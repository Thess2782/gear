34pub(crate) const RUNTIME_API_BLOCK_LIMITS_COUNT: u64 = 6;
35pub(crate) const ALLOWANCE_LIMIT_ERR: &str = "Calculation gas limit exceeded. Use your own RPC node with `--rpc-calculations-multiplier` parameter raised";

47    // Internal implementation of RPC call `gear_calculate_replyForHandle(..)`.
48    //
49    // The RPC call is used to figure out the reply that would be send
50    // on calling `Gear::send_message(..)` with following arguments.
51    pub(crate) fn calculate_reply_for_handle_impl(
52        origin: H256,
53        destination: ActorId,
54        payload: Vec<u8>,
55        gas_limit: u64,
56        value: u128,
57        allowance_multiplier: u64,
58    ) -> Result<ReplyInfo, String> {
59        // Enabling lazy-pages for this thread.
60        Self::enable_lazy_pages();
61
62        // Clearing queue.
63        QueueOf::<T>::clear();
64
65        // Calculating gas allowance for a whole operation,
66        // according to allowance multiplier.
67        let gas_allowance = allowance_multiplier.saturating_mul(BlockGasLimitOf::<T>::get());
68
69        // Updating gas allowance with calculated value.
70        Self::update_gas_allowance(gas_allowance);
71
72        // Casting types into runtime assoc-s.
73        let origin = origin.cast();
74        let value = value.unique_saturated_into();
75
76        // Preparing origin balance for extrinsic expenses.
77        let who = Self::prepare_origin_account(origin, gas_limit, value);
78
79        // Executing `send_message` call.
80        Self::send_message(who.into(), destination, payload, gas_limit, value, false)
81            .map_err(|e| Self::dispatch_err_to_string("send_message", e))?;
82
83        // Looking up queue head for message id sent above.
84        let (message_id, _) = Self::queue_head()?;
85
86        // Creating builtin dispatcher for queue processing.
87        let (builtin_dispatcher, _) = T::BuiltinDispatcherFactory::create();
88
89        // Creating new manager for queue processing.
90        let mut ext_manager = ExtManager::<T>::new(builtin_dispatcher);
91
92        // Queue processing loop.
93        //
94        // Running queue head message if exists.
95        while let Some((_, journal, _)) = Self::dequeue_head_and_run(&mut ext_manager, None)? {
96            // Looking through all notes in order to find required reply.
97            for note in &journal {
98                // Only paying attention on dispatch sends.
99                let JournalNote::SendDispatch { dispatch, .. } = note else {
100                    continue;
101                };
102
103                // Only paying attention if replies to `message_id`.
104                if let Some(code) = dispatch
105                    .reply_details()
106                    .map(ReplyDetails::into_parts)
107                    .and_then(|(replied_to, code)| replied_to.eq(&message_id).then_some(code))
108                {
109                    return Ok(ReplyInfo {
110                        payload: dispatch.payload_bytes().to_vec(),
111                        value: dispatch.value(),
112                        code,
113                    });
114                }
115            }
116
117            // Processing notes since reply wasn't found.
118            core_processor::handle_journal(journal, &mut ext_manager);
119
120            // If some message overcame block allowance, aborting processing.
121            if QueueProcessingOf::<T>::denied() {
122                return Err(ALLOWANCE_LIMIT_ERR.to_string());
123            }
124        }
125
126        // Ran out of messages in queue.
127        Err(String::from("Queue is empty, but reply wasn't found"))
128    }

130    // Internal implementation of RPC calls `gear_calculate*Entry*Gas(..)`.
131    //
132    // The RPC call is used to figure out required gas amount for successful
133    // execution of the message.
134    #[allow(clippy::too_many_arguments)]
135    pub(crate) fn calculate_gas_info_impl(
136        origin: H256,
137        kind: HandleKind,
138        initial_gas: u64,
139        payload: Vec<u8>,
140        value: u128,
141        allow_other_panics: bool,
142        allow_skip_zero_replies: bool,
143        allowance_multiplier: Option<u64>,
144    ) -> Result<GasInfo, String> {
145        // Enabling lazy-pages for this thread.
146        Self::enable_lazy_pages();
147
148        // Clearing queue.
149        QueueOf::<T>::clear();
150
151        // Calculating gas allowance for a whole operation,
152        // according to allowance multiplier.
153        let gas_allowance = allowance_multiplier
154            .unwrap_or(RUNTIME_API_BLOCK_LIMITS_COUNT)
155            .saturating_mul(BlockGasLimitOf::<T>::get());
156
157        // Updating gas allowance with calculated value.
158        Self::update_gas_allowance(gas_allowance);
159
160        // Casting types into runtime assoc-s.
161        let origin = origin.cast();
162        let value = value.unique_saturated_into();
163
164        // Preparing origin balance for extrinsic expenses.
165        let who = Self::prepare_origin_account(origin, initial_gas, value);
166
167        match kind {
168            // Executing `upload_program` call.
169            HandleKind::Init(code) => {
170                let salt = b"calculate_gas_salt".to_vec();
171
172                Self::upload_program(who.into(), code, salt, payload, initial_gas, value, false)
173                    .map_err(|e| Self::dispatch_err_to_string("upload_program", e))?;
174            }
175
176            // Executing `create_program` call.
177            HandleKind::InitByHash(code_id) => {
178                let salt = b"calculate_gas_salt".to_vec();
179
180                Self::create_program(
181                    who.into(),
182                    code_id,
183                    salt,
184                    payload,
185                    initial_gas,
186                    value,
187                    false,
188                )
189                .map_err(|e| Self::dispatch_err_to_string("create_program", e))?;
190            }
191
192            // Executing `send_message` call.
193            HandleKind::Handle(destination) => {
194                Self::send_message(who.into(), destination, payload, initial_gas, value, false)
195                    .map_err(|e| Self::dispatch_err_to_string("send_message", e))?;
196            }
197
198            // Executing `send_reply` call.
199            HandleKind::Reply(reply_to_id, _status_code) => {
200                Self::send_reply(who.into(), reply_to_id, payload, initial_gas, value, false)
201                    .map_err(|e| Self::dispatch_err_to_string("send_reply", e))?;
202            }
203
204            // Handle signal forbidden call.
205            HandleKind::Signal(_signal_from, _status_code) => {
206                return Err(String::from(
207                    "Gas calculation for `handle_signal` is not supported",
208                ));
209            }
210        };
211
212        // Looking up queue head for message id and destination sent above.
213        let (main_message_id, main_program_id) = Self::queue_head()?;
214
215        // Creating builtin dispatcher for queue processing.
216        let (builtin_dispatcher, _) = T::BuiltinDispatcherFactory::create();
217
218        // Creating new manager for queue processing.
219        let mut ext_manager = ExtManager::<T>::new(builtin_dispatcher);
220
221        // Creating forbidden funcs registry.
222        let forbidden_funcs = [SyscallName::GasAvailable];
223
224        // Getter for gas limit of the root message.
225        //
226        // For case when node is not consumed and has any (even zero) balance
227        // it means that it burned/sent all the funds and we must return it.
228        //
229        // For case when node is consumed and has zero balance it means that
230        // node moved its funds upstream to its ancestor. So this shouldn't
231        // be returned.
232        //
233        // For case when node is consumed and has non zero balance it means
234        // that it has gasless child that will consume gas further. So we
235        // handle this value as well.
236        let get_main_limit = || {
237            GasHandlerOf::<T>::get_limit(main_message_id)
238                .ok()
239                .or_else(|| {
240                    GasHandlerOf::<T>::get_limit_consumed(main_message_id)
241                        .ok()
242                        .filter(|limit| !limit.is_zero())
243                })
244        };
245
246        // Getter for identifying if message was in primary messages chain.
247        let from_main_chain = |msg_id| {
248            GasHandlerOf::<T>::get_origin_key(msg_id)
249                .map(|v| v == main_message_id.into())
250                .map_err(|_| Self::internal_err_string("Failed to get origin key"))
251        };
252
253        // Result to be returned.
254        let mut gas_info: GasInfo = Default::default();
255
256        // Queue processing loop.
257        //
258        // Running queue head message if exists.
259        while let Some((processed, journal, by_builtin)) =
260            Self::dequeue_head_and_run(&mut ext_manager, Some(forbidden_funcs.into()))?
261        {
262            // Defining if success reply was processed.
263            let success_reply = processed
264                .reply_details()
265                .map(|rd| rd.to_reply_code().is_success())
266                .unwrap_or(false);
267
268            // Extracting infallibly gas limit of processed message.
269            let gas_limit = processed.gas_limit().expect("Infallible");
270
271            // Defining if we skip checks for this message if allowed to.
272            let skip_if_allowed = !by_builtin && success_reply && gas_limit == 0;
273
274            // Looking through all notes in order to calculate gas properly.
275            for note in journal {
276                // Processing note.
277                core_processor::handle_journal(vec![note.clone()], &mut ext_manager);
278
279                // If some message overcame block allowance, aborting processing.
280                if QueueProcessingOf::<T>::denied() {
281                    return Err(ALLOWANCE_LIMIT_ERR.to_string());
282                }
283
284                // Querying gas limit of the main messages chain.
285                match get_main_limit() {
286                    // If some limit still exist, than checking the highest
287                    // diff from initial gas as calculated value.
288                    Some(remaining_gas) => {
289                        gas_info.min_limit = gas_info
290                            .min_limit
291                            .max(initial_gas.saturating_sub(remaining_gas));
292                    }
293
294                    // If limit no longer exists we need to check others for
295                    // infinite wait if they belong to main messages chain.
296                    None => {
297                        // Take into account that 'wait' syscall greedily
298                        // consumes all available gas.
299                        // Meanwhile, 'wait_for' and 'wait_up_to' should not
300                        // consume all available gas because of the limited
301                        // durations. If a duration is a big enough then it
302                        // won't matter how to calculate the limit:
303                        // it will be the same.
304                        if let JournalNote::WaitDispatch {
305                            waited_type: MessageWaitedType::Wait,
306                            ref dispatch,
307                            ..
308                        } = note
309                        {
310                            if from_main_chain(dispatch.id())? {
311                                gas_info.min_limit = initial_gas;
312                            }
313                        }
314                    }
315                }
316
317                // Parsing other types of the node for extra actions.
318                match note {
319                    // Checking sending for mailbox insertion (e.g. reserve).
320                    JournalNote::SendDispatch { dispatch, .. } => {
321                        // Extracting and casting destination to AccountId.
322                        let destination = dispatch.destination().cast();
323
324                        // Checking mailbox insertion and if newly created
325                        // dispatch is from main chain.
326                        //
327                        // NOTE: to pass `from_main_chain` call, message should
328                        // exist in system: at least in `Mailbox`.
329                        if MailboxOf::<T>::contains(&destination, &dispatch.id())
330                            && from_main_chain(dispatch.id())?
331                        {
332                            // Querying reserved balance for mailbox storing.
333                            //
334                            // NOTE: here goes extraction of the gas directly
335                            // from message, if gasless sent, than it's
336                            // queried from the storage tree.
337                            let gas_limit = dispatch
338                                .gas_limit()
339                                .or_else(|| GasHandlerOf::<T>::get_limit(dispatch.id()).ok())
340                                .ok_or_else(|| {
341                                    Self::internal_err_string(
342                                        "Failed to get gas limit after execution",
343                                    )
344                                })?;
345
346                            gas_info.reserved = gas_info.reserved.saturating_add(gas_limit);
347                        }
348                    }
349
350                    // Burning gas from main messages chain.
351                    JournalNote::GasBurned { amount, message_id } => {
352                        if from_main_chain(message_id)? {
353                            gas_info.burned = gas_info.burned.saturating_add(amount);
354                        }
355                    }
356
357                    // Checking execution for panic happened.
358                    JournalNote::MessageDispatched {
359                        outcome:
360                            CoreDispatchOutcome::MessageTrap { trap, .. }
361                            | CoreDispatchOutcome::InitFailure { reason: trap, .. },
362                        message_id,
363                        ..
364                    } if (message_id == main_message_id || !allow_other_panics)
365                        && !(skip_if_allowed && allow_skip_zero_replies) =>
366                    {
367                        return Err(format!("Program terminated with a trap: '{trap}'"));
368                    }
369
370                    _ => (),
371                }
372            }
373        }
374
375        // Defining if message is kept by waitlist.
376        gas_info.waited = WaitlistOf::<T>::contains(&main_program_id, &main_message_id);
377
378        // Returning result.
379        Ok(gas_info)
380    }

382    pub(crate) fn read_state_using_wasm_impl(
383        program_id: ActorId,
384        payload: Vec<u8>,
385        function: impl Into<String>,
386        wasm: Vec<u8>,
387        argument: Option<Vec<u8>>,
388        allowance_multiplier: Option<u64>,
389    ) -> Result<Vec<u8>, String> {
390        Self::enable_lazy_pages();
391
392        let schedule = T::Schedule::get();
393
394        if u32::try_from(wasm.len()).unwrap_or(u32::MAX) > schedule.limits.code_len {
395            return Err("Wasm too big".into());
396        }
397
398        let code = Code::try_new_mock_with_rules(
399            wasm,
400            |module| schedule.rules(module),
401            TryNewCodeConfig::new_no_exports_check(),
402        )
403        .map_err(|e| format!("Failed to construct program: {e:?}"))?;
404
405        if u32::try_from(code.code().len()).unwrap_or(u32::MAX) > schedule.limits.code_len {
406            return Err("Wasm after instrumentation too big".into());
407        }
408
409        let code_and_id = CodeAndId::new(code);
410        let code_and_id = InstrumentedCodeAndId::from(code_and_id);
411
412        let instrumented_code = code_and_id.into_parts().0;
413
414        let payload_arg = payload;
415        let mut payload = argument.unwrap_or_default();
416        payload.append(&mut Self::read_state_impl(
417            program_id,
418            payload_arg,
419            allowance_multiplier,
420        )?);
421
422        let block_info = BlockInfo {
423            height: Self::block_number().unique_saturated_into(),
424            timestamp: <pallet_timestamp::Pallet<T>>::get().unique_saturated_into(),
425        };
426
427        let gas_allowance = allowance_multiplier
428            .unwrap_or(RUNTIME_API_BLOCK_LIMITS_COUNT)
429            .saturating_mul(BlockGasLimitOf::<T>::get());
430
431        Self::update_gas_allowance(gas_allowance);
432
433        core_processor::informational::execute_for_reply::<Ext, String>(
434            function.into(),
435            instrumented_code,
436            None,
437            None,
438            payload,
439            gas_allowance,
440            block_info,
441        )
442    }

445        program_id: ActorId,
446        payload: Vec<u8>,
447        allowance_multiplier: Option<u64>,
448    ) -> Result<Vec<u8>, String> {

459        let block_info = BlockInfo {

466            .saturating_mul(BlockGasLimitOf::<T>::get());

469
470        core_processor::informational::execute_for_reply::<Ext, String>(

472            instrumented_code,
473            Some(allocations),
474            Some((program_id, memory_infix)),
475            payload,
476            gas_allowance,
477            block_info,
478        )
479    }

481    pub(crate) fn read_metahash_impl(
482        program_id: ActorId,
483        allowance_multiplier: Option<u64>,
484    ) -> Result<H256, String> {
485        Self::enable_lazy_pages();
486
487        log::debug!("Reading metahash of {program_id:?}");
488
489        let CodeWithMemoryData {
490            instrumented_code,
491            allocations,
492            memory_infix,
493        } = Self::code_with_memory(program_id)?;
494
495        let block_info = BlockInfo {
496            height: Self::block_number().unique_saturated_into(),
497            timestamp: <pallet_timestamp::Pallet<T>>::get().unique_saturated_into(),
498        };
499
500        let gas_allowance = allowance_multiplier
501            .unwrap_or(RUNTIME_API_BLOCK_LIMITS_COUNT)
502            .saturating_mul(BlockGasLimitOf::<T>::get());
503
504        Self::update_gas_allowance(gas_allowance);
505
506        core_processor::informational::execute_for_reply::<Ext, String>(
507            String::from("metahash"),
508            instrumented_code,
509            Some(allocations),
510            Some((program_id, memory_infix)),
511            Default::default(),
512            gas_allowance,
513            block_info,
514        )
515        .and_then(|bytes| {
516            H256::decode(&mut bytes.as_ref()).map_err(|_| "Failed to decode hash".into())
517        })
518    }

520    // Returns code and allocations of the given program id.
521    fn code_with_memory(program_id: ActorId) -> Result<CodeWithMemoryData, String> {
522        // Load active program from storage.
523        let program: ActiveProgram<_> = ProgramStorageOf::<T>::get_program(program_id)
524            .ok_or(String::from("Program not found"))?
525            .try_into()
526            .map_err(|e| format!("Get active program error: {e:?}"))?;
527
528        let code_id = program.code_hash.cast();
529
530        // Load instrumented binary code from storage.
531        let mut code = T::CodeStorage::get_code(code_id).ok_or_else(|| {
532            format!("Program '{program_id:?}' exists so must do code '{code_id:?}'")
533        })?;
534
535        // Reinstrument the code if necessary.
536        let schedule = T::Schedule::get();
537
538        if code.instruction_weights_version() != schedule.instruction_weights.version {
539            code = Pallet::<T>::reinstrument_code(code_id, &schedule)
540                .map_err(|e| format!("Code {code_id:?} failed reinstrumentation: {e:?}"))?;
541        }
542
543        let allocations = ProgramStorageOf::<T>::allocations(program_id).unwrap_or_default();
544
545        Ok(CodeWithMemoryData {
546            instrumented_code: code,
547            allocations,
548            memory_infix: program.memory_infix,
549        })
550    }

552    // Prepares account id to be able to execute some extrinsic in terms of funds.
553    fn prepare_origin_account(
554        origin: AccountIdOf<T>,
555        gas: u64,
556        value: BalanceOf<T>,
557    ) -> RawOrigin<AccountIdOf<T>> {
558        // Querying transferrable balance of the origin taking into account a possibility of
559        // a part of its `free` balance being `frozen`.
560        let origin_balance = <CurrencyOf<T> as fungible::Inspect<_>>::reducible_balance(
561            &origin,
562            Preservation::Expendable,
563            Fortitude::Polite,
564        );
565
566        // Calculating amount of value to be paid for gas.
567        let value_for_gas = <T as pallet_gear_bank::Config>::GasMultiplier::get().gas_to_value(gas);
568
569        // Required balance of the account.
570        let required_balance = CurrencyOf::<T>::minimum_balance()
571            .saturating_add(value_for_gas)
572            .saturating_add(value);
573
574        // Updating balance of the account.
575        let _ = CurrencyOf::<T>::deposit_creating(
576            &origin,
577            required_balance.saturating_sub(origin_balance),
578        );
579
580        // Returning origin account as signed origin.
581        RawOrigin::Signed(origin)
582    }

584    // Returns none if queue is empty, otherwise - processed message,
585    // resulting journal notes of the processing and bool, defining
586    // was it processed by builtin actor or not.
587    fn dequeue_head_and_run(
588        ext_manager: &mut ExtManager<T>,
589        forbidden_funcs: Option<BTreeSet<SyscallName>>,
590    ) -> Result<Option<(Dispatch, Vec<JournalNote>, bool)>, String> {
591        // Extracting queued dispatch.
592        let head =
593            QueueOf::<T>::dequeue().map_err(|_| Self::internal_err_string("Queue corrupted"))?;
594
595        let Some(dispatch) = head else {
596            return Ok(None);
597        };
598
599        // Extracting destination from dispatch.
600        let destination = dispatch.destination();
601
602        // Querying gas limit for dispatch.
603        let gas_limit = GasHandlerOf::<T>::get_limit(dispatch.id())
604            .map_err(|_| Self::internal_err_string("Failed to get gas limit"))?;
605
606        // Storing processed dispatch.
607        let processed = Dispatch::new(
608            dispatch.kind(),
609            Message::new(
610                dispatch.id(),
611                dispatch.source(),
612                dispatch.destination(),
613                dispatch
614                    .payload_bytes()
615                    .to_vec()
616                    .try_into()
617                    .expect("Infallible"),
618                Some(gas_limit),
619                dispatch.value(),
620                dispatch.details(),
621            ),
622        );
623
624        // Processing of the message, if destination is builtin actor.
625        let builtin_dispatcher = ext_manager.builtins();
626        if let Some(info) = builtin_dispatcher.lookup(&destination) {
627            let journal = builtin_dispatcher.run(info, dispatch, gas_limit);
628            return Ok(Some((processed, journal, true)));
629        }
630
631        let mut block_config = Self::block_config();
632
633        if let Some(forbidden_funcs) = forbidden_funcs {
634            block_config.forbidden_funcs = forbidden_funcs;
635        }
636
637        // Program's balance that it can spend during a message execution.
638        let disposable_balance = <CurrencyOf<T> as fungible::Inspect<_>>::reducible_balance(
639            &destination.cast(),
640            Preservation::Expendable,
641            Fortitude::Polite,
642        );
643
644        // Processing of the message, if destination is common program.
645        let journal = Self::run_queue_step(QueueStep {
646            block_config: &block_config,
647            gas_limit,
648            dispatch,
649            balance: disposable_balance.unique_saturated_into(),
650        });
651
652        Ok(Some((processed, journal, false)))
653    }

655    // Converts given dispatch error into dedicated runtime api string format.
656    fn dispatch_err_to_string(
657        extrinsic_name: &'static str,
658        e: DispatchErrorWithPostInfo<PostDispatchInfo>,
659    ) -> String {
660        // Extracting index of module returned error, if possible.
661        let error_module_idx = match e.error {
662            DispatchError::Module(ModuleError { index, .. }) => Some(index as usize),
663            _ => None,
664        };
665
666        // Converting dispatch error into string representation in default impl.
667        let error_message: &'static str = e.into();
668
669        // Creating result message.
670        let mut res = format!("Extrinsic `gear.{extrinsic_name}` failed: '{error_message}'");
671
672        // Extracting `pallet_gear` index from runtime to compare with dispatch error.
673        let Some(gear_module_idx) = PalletInfoOf::<T>::index::<Self>() else {
674            return Self::internal_err_string("No index found for `pallet_gear` in the runtime");
675        };
676
677        // Appending result message with pallet index returned error, if not this.
678        if let Some(module_idx) = error_module_idx.filter(|i| *i != gear_module_idx) {
679            res = format!("{res} (pallet index of the error: {module_idx}");
680        }
681
682        res
683    }
684
685    // Queries first element of the queue and extracts its message id and destination.
686    fn queue_head() -> Result<(MessageId, ActorId), String> {
687        QueueOf::<T>::iter()
688            .next()
689            .ok_or_else(|| Self::internal_err_string("Failed to get last message from the queue"))
690            .and_then(|queued| {
691                queued
692                    .map(|dispatch| (dispatch.id(), dispatch.destination()))
693                    .map_err(|_| Self::internal_err_string("Failed to extract queued dispatch"))
694            })
695    }
696
697    // Updates gas allowance and allows queue processing.
698    fn update_gas_allowance(gas_allowance: u64) {
699        GasAllowanceOf::<T>::put(gas_allowance);
700        QueueProcessingOf::<T>::allow();
701    }

704    fn internal_err_string(message: impl ToString) -> String {
705        format!(
706            "Internal error: entered unreachable code '{}'",
707            message.to_string()
708        )
709    }
