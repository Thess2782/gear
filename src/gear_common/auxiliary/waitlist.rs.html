19//! Auxiliary implementation of the waitlist.
20
21use super::{AuxiliaryDoubleStorageWrap, BlockNumber, DoubleBTreeMap};
22use crate::storage::{Interval, WaitlistError, WaitlistImpl, WaitlistKeyGen};
23use core::cell::RefCell;
24use gear_core::{
25    ids::{ActorId, MessageId},
26    message::StoredDispatch,
27};
28
29/// Waitlist implementation that can be used in a native, non-wasm runtimes.
30pub type AuxiliaryWaitlist<WaitListCallbacks> = WaitlistImpl<
31    WaitlistStorageWrap,
32    WaitlistedMessage,
33    BlockNumber,
34    WaitlistErrorImpl,
35    WaitlistErrorImpl,
36    WaitListCallbacks,
37    WaitlistKeyGen,
38>;
39/// Type represents message stored in the waitlist.
40pub type WaitlistedMessage = StoredDispatch;
41
42std::thread_local! {
43    // Definition of the waitlist (`StorageDoubleMap`) global storage, accessed by the `Waitlist` trait implementor.
44    pub(crate) static WAITLIST_STORAGE: RefCell<DoubleBTreeMap<ActorId, MessageId, (WaitlistedMessage, Interval<BlockNumber>)>> = const { RefCell::new(DoubleBTreeMap::new()) };
45}
46
47/// `Waitlist` double storage map manager.
48pub struct WaitlistStorageWrap;
49
50impl AuxiliaryDoubleStorageWrap for WaitlistStorageWrap {
51    type Key1 = ActorId;
52    type Key2 = MessageId;
53    type Value = (WaitlistedMessage, Interval<BlockNumber>);
54
55    fn with_storage<F, R>(f: F) -> R
56    where
57        F: FnOnce(&DoubleBTreeMap<Self::Key1, Self::Key2, Self::Value>) -> R,
58    {
59        WAITLIST_STORAGE.with_borrow(f)
60    }
61
62    fn with_storage_mut<F, R>(f: F) -> R
63    where
64        F: FnOnce(&mut DoubleBTreeMap<Self::Key1, Self::Key2, Self::Value>) -> R,
65    {
66        WAITLIST_STORAGE.with_borrow_mut(f)
67    }
68}
69
70/// An implementor of the error returned from calling `Waitlist` trait functions
71#[derive(Debug)]
72pub enum WaitlistErrorImpl {
73    DuplicateKey,
74    ElementNotFound,
75}
76
77impl WaitlistError for WaitlistErrorImpl {
78    fn duplicate_key() -> Self {
79        Self::DuplicateKey
80    }
81
82    fn element_not_found() -> Self {
83        Self::ElementNotFound
84    }
85}
