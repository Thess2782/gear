32/// Signal handler which has been set before lazy-pages initialization.
33/// Currently use to support wasmer signal handler.
34/// Wasmer protects memory around wasm memory and for stack limits.
35/// It makes it only in `store` initialization when executor is created,
36/// see https://github.com/gear-tech/substrate/blob/gear-stable/client/executor/common/src/sandbox/wasmer_backend.rs
37/// and https://github.com/wasmerio/wasmer/blob/e6857d116134bdc9ab6a1dabc3544cf8e6aee22b/lib/vm/src/trap/traphandlers.rs#L548
38/// So, if we receive signal from unknown memory we should try to use old (wasmer) signal handler.
39static OLD_SIG_HANDLER: OnceLock<SigHandler> = OnceLock::new();
40
41cfg_if! {
42    if #[cfg(all(target_os = "linux", target_arch = "x86_64"))] {
43        unsafe fn ucontext_get_write(ucontext: *mut nix::libc::ucontext_t) -> Option<bool> {
44            let error_reg = nix::libc::REG_ERR as usize;
45            let error_code = unsafe { *ucontext }.uc_mcontext.gregs[error_reg];
46            // Use second bit from err reg. See https://git.io/JEQn3
47            Some(error_code & 0b10 == 0b10)
48        }
49    } else if #[cfg(all(target_os = "macos", target_arch = "x86_64"))] {
50        unsafe fn ucontext_get_write(ucontext: *mut nix::libc::ucontext_t) -> Option<bool> {
51            // See https://wiki.osdev.org/Exceptions
52            const WRITE_BIT_MASK: u32 = 0b10;
53            const TRAPNO: u16 = 0xe; // Page Fault
54
55            let mcontext = unsafe { *ucontext }.uc_mcontext;
56            let exception_state = unsafe { *mcontext }.__es;
57            let trapno = exception_state.__trapno;
58            let err = exception_state.__err;
59
60            assert_eq!(trapno, TRAPNO);
61
62            Some(err & WRITE_BIT_MASK == WRITE_BIT_MASK)
63        }
64    } else if #[cfg(all(target_os = "macos", target_arch = "aarch64"))] {
65        unsafe fn ucontext_get_write(ucontext: *mut nix::libc::ucontext_t) -> Option<bool> {
66            // See https://developer.arm.com/documentation/ddi0595/2021-03/AArch64-Registers/ESR-EL1--Exception-Syndrome-Register--EL1-
67            const WNR_BIT_MASK: u32 = 0b100_0000; // Write not Read bit
68            const EXCEPTION_CLASS_SHIFT: u32 = u32::BITS - 6;
69            const EXCEPTION_CLASS: u32 = 0b10_0100; // Data Abort from a lower Exception Level
70
71            let ucontext = unsafe { ucontext.as_mut() }?;
72            let mcontext = ucontext.uc_mcontext;
73            let exception_state = unsafe { *mcontext }.__es;
74            let esr = exception_state.__esr;
75
76            let exception_class = esr >> EXCEPTION_CLASS_SHIFT;
77            assert_eq!(exception_class, EXCEPTION_CLASS);
78
79            Some(esr & WNR_BIT_MASK == WNR_BIT_MASK)
80        }
81    } else {
82        compile_error!("lazy-pages are not supported on your system. Disable `lazy-pages` feature");
83    }
84}

86extern "C" fn handle_sigsegv<H>(sig: i32, info: *mut siginfo_t, ucontext: *mut c_void)
87where
88    H: UserSignalHandler,
89{
90    unsafe {
91        let addr = (*info).si_addr();
92        let is_write = ucontext_get_write(ucontext as *mut _);
93        let exc_info = ExceptionInfo {
94            fault_addr: addr as *mut _,
95            is_write,
96        };
97
98        if let Err(err) = H::handle(exc_info) {
99            let old_sig_handler_works = match err {
100                Error::OutOfWasmMemoryAccess | Error::WasmMemAddrIsNotSet => {
101                    old_sig_handler(sig, info, ucontext)
102                }
103                _ => false,
104            };
105            if !old_sig_handler_works {
106                panic!("Signal handler failed: {err}");
107            }
108        }
109    }
110}

114#[derive(Debug, Clone, Copy, derive_more::Display)]
115enum ThreadInitError {
116    #[display("Cannot get information about old signal stack: {_0}")]
117    OldStack(Errno),
118    #[display("Cannot mmap space for signal stack: {_0}")]
119    Mmap(Errno),
120    #[display("Cannot set new signal stack: {_0}")]
121    SigAltStack(Errno),
122}
123
124fn init_for_thread_internal() -> Result<(), ThreadInitError> {
125    use core::{mem, ptr};
126
127    // Should be enough for lazy-pages signal handler.
128    // Equal to libc::SIGSTKSZ on macos M1.
129    const SIGNAL_STACK_SIZE: usize = 0x20000;
130
131    enum StackInfo {
132        UseOldStack,
133        NewStack(*mut libc::c_void),
134    }
135
136    impl Drop for StackInfo {
137        fn drop(&mut self) {
138            if let StackInfo::NewStack(mmap_ptr) = self {
139                unsafe {
140                    // Deallocate the stack memory.
141                    if libc::munmap(*mmap_ptr, SIGNAL_STACK_SIZE) != 0 {
142                        log::error!(
143                            "Cannot deallocate signal stack memory during the thread shutdown: {}",
144                            errno::errno()
145                        );
146                    }
147                }
148            }
149        }
150    }
151
152    unsafe fn init_sigstack() -> Result<StackInfo, ThreadInitError> {
153        // Check whether old signal stack exist and suitable for lazy-pages signal handler.
154        let mut old_stack = unsafe { mem::zeroed() };
155        let res = unsafe { libc::sigaltstack(ptr::null(), &mut old_stack) };
156        if res != 0 {
157            return Err(ThreadInitError::OldStack(errno::errno()));
158        }
159        if old_stack.ss_flags & libc::SS_DISABLE == 0 && old_stack.ss_size >= SIGNAL_STACK_SIZE {
160            return Ok(StackInfo::UseOldStack);
161        }
162
163        // Alloc memory for new signal stack.
164        let ptr = unsafe {
165            libc::mmap(
166                ptr::null_mut(),
167                SIGNAL_STACK_SIZE,
168                libc::PROT_READ | libc::PROT_WRITE,
169                libc::MAP_PRIVATE | libc::MAP_ANON,
170                -1,
171                0,
172            )
173        };
174        if ptr == libc::MAP_FAILED {
175            return Err(ThreadInitError::Mmap(errno::errno()));
176        }
177
178        // Mark allocated memory as new signal stack.
179        let new_stack = libc::stack_t {
180            ss_sp: ptr,
181            ss_flags: 0,
182            ss_size: SIGNAL_STACK_SIZE,
183        };
184        let res = unsafe { libc::sigaltstack(&new_stack, ptr::null_mut()) };
185        if res != 0 {
186            return Err(ThreadInitError::SigAltStack(errno::errno()));
187        }
188
189        log::debug!(
190            "Set new signal stack: ptr = {:?}, size = {:#x}",
191            ptr,
192            SIGNAL_STACK_SIZE
193        );
194
195        Ok(StackInfo::NewStack(ptr))
196    }
197
198    thread_local! {
199        static TLS: Result<StackInfo, ThreadInitError> = unsafe { init_sigstack() };
200    }
201
202    TLS.with(|tls| tls.as_ref().map(|_| ()).map_err(|err| *err))
203}

205pub(crate) unsafe fn init_for_thread() -> Result<(), String> {
206    init_for_thread_internal().map_err(|err| err.to_string())
207}

209pub(crate) unsafe fn setup_signal_handler<H>() -> io::Result<()>
210where
211    H: UserSignalHandler,
212{
213    let handler = signal::SigHandler::SigAction(handle_sigsegv::<H>);
214    // Set additional SA_ONSTACK and SA_NODEFER to avoid problems with wasmer executor.
215    // See comment from shorturl.at/KMO68 :
216    // ```
217    //  SA_ONSTACK allows us to handle signals on an alternate stack,
218    //  so that the handler can run in response to running out of
219    //  stack space on the main stack. Rust installs an alternate
220    //  stack with sigaltstack, so we rely on that.
221    //  SA_NODEFER allows us to reenter the signal handler if we
222    //  crash while handling the signal, and fall through to the
223    //  Breakpad handler by testing handlingSegFault.
224    // ```
225    let sig_action = signal::SigAction::new(
226        handler,
227        signal::SaFlags::SA_SIGINFO | signal::SaFlags::SA_ONSTACK | signal::SaFlags::SA_NODEFER,
228        signal::SigSet::empty(),
229    );
230
231    let signal = if cfg!(target_os = "macos") {
232        signal::SIGBUS
233    } else {
234        signal::SIGSEGV
235    };
236
237    let old_sigaction =
238        unsafe { signal::sigaction(signal, &sig_action) }.map_err(io::Error::from)?;
239    let handler = old_sigaction.handler();
240    let _ = OLD_SIG_HANDLER
241        .set(handler)
242        .map(|_| log::trace!("Save old signal handler: {:?}", handler));
243
244    Ok(())
245}

247unsafe fn old_sig_handler(sig: i32, info: *mut siginfo_t, ucontext: *mut c_void) -> bool {
248    if let Some(old_sig_handler) = OLD_SIG_HANDLER.get() {
249        match old_sig_handler {
250            SigHandler::SigDfl | SigHandler::SigIgn => false,
251            SigHandler::Handler(func) => {
252                func(sig);
253                true
254            }
255            SigHandler::SigAction(func) => {
256                func(sig, info, ucontext);
257                true
258            }
259        }
260    } else {
261        false
262    }
