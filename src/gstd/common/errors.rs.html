33/// `Result` type with a predefined error type ([`Error`]).
34pub type Result<T, E = Error> = core::result::Result<T, E>;
35
36/// Common error type returned by API functions from other modules.
37#[derive(Clone, Debug, Eq, PartialEq)]
38pub enum Error {
39    /* Protocol under-hood errors */
40    /// Error type from `gcore`.
41    ///
42    /// NOTE: this error could only be returned from syscalls.
43    Core(CoreError),
44
45    /* API lib under-hood errors */
46    /// Conversion error.
47    ///
48    /// NOTE: this error returns from incorrect bytes conversion.
49    Convert(ConversionError),
50
51    /// `scale-codec` decoding error.
52    ///
53    /// NOTE: this error returns from APIs that return specific `Decode` types.
54    Decode(CodecError),
55
56    /// Gstd API usage error.
57    ///
58    /// Note: this error returns from `gstd` APIs in case of invalid arguments.
59    Gstd(UsageError),
60
61    /* Business logic errors */
62    /// Received error reply while awaited response from another actor.
63    ///
64    /// NOTE: this error could only be returned from async messaging.
65    // TODO: consider to load payload lazily (#4595)
66    ErrorReply(ErrorReplyPayload, ErrorReplyReason),
67
68    /// Received reply that couldn't be identified as successful or not
69    /// due to unsupported reply code.
70    ///
71    /// NOTE: this error could only be returned from async messaging.
72    UnsupportedReply(Vec<u8>),
73
74    /// Timeout reached while expecting for reply.
75    ///
76    /// NOTE: this error could only be returned from async messaging.
77    Timeout(u32, u32),
78}
79
80impl Error {
81    /// Check whether an error is [`Error::Timeout`].
82    pub fn timed_out(&self) -> bool {
83        matches!(self, Error::Timeout(..))
84    }
85
86    /// Check whether an error is [`SimpleExecutionError::UserspacePanic`] from
87    /// error reply and return its decoded message's payload of
88    /// a custom type.
89    pub fn error_reply_panic<T: Decode>(&self) -> Option<Result<T, Self>> {
90        self.error_reply_panic_bytes()
91            .map(|mut bytes| T::decode(&mut bytes).map_err(Error::Decode))
92    }
93
94    /// Check whether an error is [`SimpleExecutionError::UserspacePanic`] from
95    /// error reply and return its payload as bytes.
96    pub fn error_reply_panic_bytes(&self) -> Option<&[u8]> {
97        if let Self::ErrorReply(
98            payload,
99            ErrorReplyReason::Execution(SimpleExecutionError::UserspacePanic),
100        ) = self
101        {
102            Some(&payload.0)
103        } else {
104            None
105        }
106    }
107
108    /// Check whether an error is [`SimpleUnavailableActorError::ProgramExited`]
109    /// from error reply and return inheritor of exited program.
110    pub fn error_reply_exit_inheritor(&self) -> Option<ActorId> {
111        if let Self::ErrorReply(
112            payload,
113            ErrorReplyReason::UnavailableActor(SimpleUnavailableActorError::ProgramExited),
114        ) = self
115        {
116            let id = ActorId::try_from(payload.0.as_slice())
117                .unwrap_or_else(|e| unreachable!("protocol always returns valid `ActorId`: {e}"));
118            Some(id)
119        } else {
120            None
121        }
122    }
123}
124
125impl fmt::Display for Error {
126    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
127        match self {
128            Error::Core(e) => fmt::Display::fmt(e, f),
129            Error::Convert(e) => write!(f, "Conversion error: {e:?}"),
130            Error::Decode(e) => write!(f, "Scale codec decoding error: {e}"),
131            Error::Gstd(e) => write!(f, "`Gstd` API error: {e:?}"),
132            Error::ErrorReply(err, reason) => write!(f, "Received reply '{err}' due to {reason:?}"),
133            Error::UnsupportedReply(payload) => {
134                write!(f, "Received unsupported reply '0x{}'", hex::encode(payload))
135            }
136            Error::Timeout(expected, now) => {
137                write!(f, "Timeout has occurred: expected at {expected}, now {now}")
138            }
139        }
140    }
141}
142
143impl From<CoreError> for Error {
144    fn from(err: CoreError) -> Self {
145        Self::Core(err)
146    }
147}
148
149impl From<ConversionError> for Error {
150    fn from(err: ConversionError) -> Self {
151        Self::Convert(err)
152    }
153}

155/// New-type representing error reply payload. Expected to be utf-8 string.
156#[derive(Clone, Eq, PartialEq)]
157pub struct ErrorReplyPayload(pub Vec<u8>);
158
159impl ErrorReplyPayload {
160    /// Returns byte slice.
161    pub fn as_bytes(&self) -> &[u8] {
162        self.0.as_slice()
163    }
164
165    /// Represents self as utf-8 str, if possible.
166    pub fn try_as_str(&self) -> Option<&str> {
167        str::from_utf8(&self.0).ok()
168    }
169
170    /// Returns inner byte vector.
171    pub fn into_inner(self) -> Vec<u8> {
172        self.0
173    }
174}
175
176impl From<Vec<u8>> for ErrorReplyPayload {
177    fn from(value: Vec<u8>) -> Self {
178        Self(value)
179    }
180}
181
182impl fmt::Debug for ErrorReplyPayload {
183    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
184        self.try_as_str()
185            .map(|v| write!(f, "{v}"))
186            .unwrap_or_else(|| write!(f, "0x{}", hex::encode(&self.0)))
187    }
188}
189
190impl fmt::Display for ErrorReplyPayload {
191    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
192        fmt::Debug::fmt(self, f)
193    }
194}

196/// Error type returned by gstd API while using invalid arguments.
197#[derive(Clone, Debug, Eq, PartialEq)]
198pub enum UsageError {
199    /// This error occurs when providing zero duration to waiting functions
200    /// (e.g. see `exactly` and `up_to` functions in
201    /// [`CodecMessageFuture`](crate::msg::CodecMessageFuture)).
202    EmptyWaitDuration,
203    /// This error occurs when providing zero gas amount to system gas reserving
204    /// function (see
205    /// [`Config::set_system_reserve`](crate::Config::set_system_reserve)).
206    ZeroSystemReservationAmount,
207    /// This error occurs when providing zero duration to mutex lock function
208    ZeroMxLockDuration,
209    /// This error occurs when handle_reply is called without (or with zero)
210    /// reply deposit
211    /// (see [`MessageFuture::handle_reply`](crate::msg::MessageFuture::handle_reply)).
212    ZeroReplyDeposit,
213}
214
215impl fmt::Display for UsageError {
216    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
217        match self {
218            UsageError::EmptyWaitDuration => write!(f, "Wait duration can not be zero"),
219            UsageError::ZeroSystemReservationAmount => {
220                write!(f, "System reservation amount can not be zero in config")
221            }
222            UsageError::ZeroMxLockDuration => write!(f, "Mutex lock duration can not be zero"),
223            UsageError::ZeroReplyDeposit => {
224                write!(f, "Reply deposit can not be zero when setting reply hook")
225            }
226        }
227    }
