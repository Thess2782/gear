36/// Limited len vector.
37/// `T` is data type.
38/// `E` is overflow error type.
39/// `N` is max len which a vector can have.
40#[derive(Clone, Default, Eq, Hash, Ord, PartialEq, PartialOrd, Decode, Encode, TypeInfo)]
41#[cfg_attr(feature = "std", derive(serde::Serialize, serde::Deserialize))]
42pub struct LimitedVec<T, E, const N: usize>(Vec<T>, PhantomData<E>);
43
44/// Formatter for [`LimitedVec`] will print to precision of 8 by default, to print the whole data, use `{:+}`.
45impl<T: Clone + Default, E: Default, const N: usize> Display for LimitedVec<T, E, N>
46where
47    [T]: AsRef<[u8]>,
48{
49    fn fmt(&self, f: &mut Formatter<'_>) -> fmt::Result {
50        let len = self.0.len();
51        let median = len.div_ceil(2);
52
53        let mut e1 = median;
54        let mut s2 = median;
55
56        if let Some(precision) = f.precision() {
57            if precision < median {
58                e1 = precision;
59                s2 = len - precision;
60            }
61        } else if !f.sign_plus() && median > 8 {
62            e1 = 8;
63            s2 = len - 8;
64        }
65
66        let p1 = hex::encode(&self.0[..e1]);
67        let p2 = hex::encode(&self.0[s2..]);
68        let sep = if e1.ne(&s2) { ".." } else { Default::default() };
69
70        if f.alternate() {
71            write!(f, "LimitedVec(0x{p1}{sep}{p2})")
72        } else {
73            write!(f, "0x{p1}{sep}{p2}")
74        }
75    }
76}

87impl<T: Clone, E: Default, const N: usize> TryFrom<&[T]> for LimitedVec<T, E, N> {
88    type Error = E;
89    fn try_from(x: &[T]) -> Result<Self, Self::Error> {
90        (x.len() <= N).then_some(()).ok_or_else(E::default)?;
91        Ok(Self(Vec::from(x), PhantomData))
92    }
93}

95impl<T, E: Default, const N: usize> TryFrom<Vec<T>> for LimitedVec<T, E, N> {
96    type Error = E;
97    fn try_from(x: Vec<T>) -> Result<Self, Self::Error> {
98        (x.len() <= N).then_some(()).ok_or_else(E::default)?;
99        Ok(Self(x, PhantomData))
100    }
101}

103impl<T: Clone + Default, E: Default, const N: usize> LimitedVec<T, E, N> {
104    /// Maximum length of the vector.
105    pub const MAX_LEN: usize = N;
106
107    /// Constructs a new, empty `LimitedVec<T>`.
108    pub const fn new() -> Self {
109        Self(Vec::new(), PhantomData)
110    }
111
112    /// Tries to create new limited vector of length `len`
113    /// with default initialized elements.
114    pub fn try_new_default(len: usize) -> Result<Self, E> {
115        (len <= N).then_some(()).ok_or_else(E::default)?;
116        Ok(Self(vec![T::default(); len], PhantomData))
117    }
118
119    /// Creates new limited vector with default initialized elements.
120    pub fn new_default() -> Self {
121        Self(vec![T::default(); N], PhantomData)
122    }
123
124    /// Creates limited vector filled with the specified `value`.
125    pub fn filled_with(value: T) -> Self {
126        Self(vec![value; N], PhantomData)
127    }
128
129    /// Extends the array to its limit and fills with the specified `value`.
130    pub fn extend_with(&mut self, value: T) {
131        self.0.resize(N, value);
132    }
133
134    /// Append `value` to the end of vector.
135    pub fn try_push(&mut self, value: T) -> Result<(), E> {
136        (self.0.len() != N).then_some(()).ok_or_else(E::default)?;
137        self.0.push(value);
138        Ok(())
139    }
140
141    /// Append `values` to the end of vector.
142    pub fn try_extend_from_slice(&mut self, values: &[T]) -> Result<(), E> {
143        self.0
144            .len()
145            .checked_add(values.len())
146            .and_then(|len| (len <= N).then_some(()))
147            .ok_or_else(E::default)?;
148
149        self.0.extend_from_slice(values);
150
151        Ok(())
152    }
153
154    /// Append `values` to the begin of vector.
155    pub fn try_prepend(&mut self, values: Self) -> Result<(), E> {
156        self.0
157            .len()
158            .checked_add(values.0.len())
159            .and_then(|len| (len <= N).then_some(()))
160            .ok_or_else(E::default)?;
161
162        self.0.splice(0..0, values.0);
163
164        Ok(())
165    }
166
167    /// Returns ref to the internal data.
168    pub fn inner(&self) -> &[T] {
169        &self.0
170    }
171
172    /// Returns mut ref to the internal data slice.
173    pub fn inner_mut(&mut self) -> &mut [T] {
174        &mut self.0
175    }
176
177    /// Clones self into vector.
178    pub fn to_vec(&self) -> Vec<T> {
179        self.0.clone()
180    }
181
182    /// Destruct limited vector and returns inner vector.
183    pub fn into_vec(self) -> Vec<T> {
184        self.0
185    }
186
187    /// Returns max len which this type of limited vector can have.
188    pub const fn max_len() -> usize {
189        N
190    }
191}

193/// Max memory size, which runtime can allocate at once.
194/// Substrate allocator restrict allocations bigger then 512 wasm pages at once.
195/// See more information about:
196/// https://github.com/paritytech/substrate/blob/cc4d5cc8654d280f03a13421669ba03632e14aa7/client/allocator/src/freeing_bump.rs#L136-L149
197/// https://github.com/paritytech/substrate/blob/cc4d5cc8654d280f03a13421669ba03632e14aa7/primitives/core/src/lib.rs#L385-L388
198const RUNTIME_MAX_ALLOC_SIZE: usize = 512 * 0x10000;
199
200/// Take half from [RUNTIME_MAX_ALLOC_SIZE] in order to avoid problems with capacity overflow.
201const RUNTIME_MAX_BUFF_SIZE: usize = RUNTIME_MAX_ALLOC_SIZE / 2;
202
203/// Runtime buffer size exceed error
204#[derive(
205    Clone, Copy, Default, Debug, Eq, Hash, Ord, PartialEq, PartialOrd, Decode, Encode, TypeInfo,
206)]
207pub struct RuntimeBufferSizeError;
208
209impl From<RuntimeBufferSizeError> for &str {
210    fn from(_: RuntimeBufferSizeError) -> Self {
211        "Runtime buffer size exceed"
212    }
213}
214
215impl Display for RuntimeBufferSizeError {
216    fn fmt(&self, f: &mut Formatter<'_>) -> fmt::Result {
217        f.write_str((*self).into())
218    }
219}
220
221/// Buffer which size cannot be bigger then max allowed allocation size in runtime.
222pub type RuntimeBuffer = LimitedVec<u8, RuntimeBufferSizeError, RUNTIME_MAX_BUFF_SIZE>;
223
224/// Max payload size which one message can have (8 MiB).
225pub const MAX_PAYLOAD_SIZE: usize = 8 * 1024 * 1024;
226
227// **WARNING**: do not remove this check
228const _: () = assert!(MAX_PAYLOAD_SIZE <= u32::MAX as usize);
229
230/// Payload size exceed error
231#[derive(
232    Clone, Copy, Default, Debug, Eq, Hash, Ord, PartialEq, PartialOrd, Decode, Encode, TypeInfo,
233)]
234#[cfg_attr(feature = "std", derive(serde::Serialize, serde::Deserialize))]
235pub struct PayloadSizeError;
236
237impl From<PayloadSizeError> for &str {
238    fn from(_: PayloadSizeError) -> Self {
239        "Payload size limit exceeded"
240    }
241}
242
243impl Display for PayloadSizeError {
244    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
245        f.write_str((*self).into())
246    }
247}
248
249/// Payload type for message.
250pub type Payload = LimitedVec<u8, PayloadSizeError, MAX_PAYLOAD_SIZE>;
251
252impl Payload {
253    /// Get payload length as u32.
254    pub fn len_u32(&self) -> u32 {
255        // Safe, cause it's guarantied: `MAX_PAYLOAD_SIZE` <= u32::MAX
256        self.inner().len() as u32
257    }
258}
259
260impl MaxEncodedLen for Payload {
261    fn max_encoded_len() -> usize {
262        Compact::<u32>::max_encoded_len() + MAX_PAYLOAD_SIZE
263    }
264}

266/// Panic buffer which size cannot be bigger then max allowed payload size.
267#[derive(
268    Clone,
269    Default,
270    Eq,
271    Hash,
272    Ord,
273    PartialEq,
274    PartialOrd,
275    Decode,
276    Encode,
277    TypeInfo,
278    derive_more::From,
279    derive_more::Into,
280)]
281#[cfg_attr(feature = "std", derive(serde::Serialize, serde::Deserialize))]
282pub struct PanicBuffer(Payload);
283
284impl PanicBuffer {
285    /// Returns ref to the internal data.
286    pub fn inner(&self) -> &Payload {
287        &self.0
288    }
289
290    fn to_limited_str(&self) -> Option<LimitedStr> {
291        let s = core::str::from_utf8(self.0.inner()).ok()?;
292        LimitedStr::try_from(s).ok()
293    }
294}
295
296impl From<LimitedStr<'_>> for PanicBuffer {
297    fn from(value: LimitedStr) -> Self {
298        const _: () = assert!(crate::str::TRIMMED_MAX_LEN <= MAX_PAYLOAD_SIZE);
299        Payload::try_from(value.into_inner().into_owned().into_bytes())
300            .map(Self)
301            .unwrap_or_else(|PayloadSizeError| {
302                unreachable!("`LimitedStr` is always smaller than maximum payload size",)
303            })
304    }
305}
306
307impl Display for PanicBuffer {
308    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
309        if let Some(s) = self.to_limited_str() {
310            Display::fmt(&s, f)
311        } else {
312            Display::fmt(&self.0, f)
313        }
314    }
315}
316
317impl Debug for PanicBuffer {
318    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
319        if let Some(s) = self.to_limited_str() {
320            Debug::fmt(s.as_str(), f)
321        } else {
322            Debug::fmt(&self.0, f)
323        }
324    }
325}

333    const N: usize = 1000;

338    #[test]
339    fn test_try_from() {
340        let v1 = vec![1; N];
341        let v2 = vec![1; N + 1];
342        let v3 = vec![1; N - 1];
343
344        let x = TestBuffer::try_from(v1).unwrap();
345        let _ = TestBuffer::try_from(v2).expect_err("Must be err because of size overflow");
346        let z = TestBuffer::try_from(v3).unwrap();
347
348        assert_eq!(x.inner().len(), N);
349        assert_eq!(z.inner().len(), N - 1);
350        assert_eq!(x.inner()[N / 2], 1);
351        assert_eq!(z.inner()[N / 2], 1);
352    }
353
354    #[test]
355    fn test_new_default() {
356        let x = LimitedVec::<String, RuntimeBufferSizeError, N>::try_new_default(N).unwrap();
357        assert!(
358            LimitedVec::<u64, RuntimeBufferSizeError, N>::try_new_default(N + 1).is_err(),
359            "Must be error because of size overflow"
360        );
361        let z = LimitedVec::<Vec<u8>, RuntimeBufferSizeError, N>::try_new_default(0).unwrap();
362
363        assert_eq!(x.inner().len(), N);
364        assert_eq!(z.inner().len(), 0);
365        assert_eq!(x.inner()[N / 2], "");
366    }
367
368    #[test]
369    fn test_prepend_works() {
370        let mut buf = TestBuffer::try_from(vec![1, 2, 3, 4, 5]).unwrap();
371        let prepend_buf = TestBuffer::try_from(vec![6, 7, 8]).unwrap();
372        buf.try_prepend(prepend_buf).unwrap();
373
374        assert_eq!(buf.inner(), &[6, 7, 8, 1, 2, 3, 4, 5]);
375    }

377    #[test]
378    fn test_full() {
379        let mut x = TestBuffer::try_from(vec![1; N]).unwrap();
380        let mut y = TestBuffer::try_from(vec![2; N / 2]).unwrap();
381        let mut z = TestBuffer::try_from(vec![3; 0]).unwrap();
382
383        x.try_push(42).unwrap_err();
384        y.try_push(42).unwrap();
385        z.try_push(42).unwrap();
386
387        x.try_extend_from_slice(&[1, 2, 3]).unwrap_err();
388        y.try_extend_from_slice(&[1, 2, 3]).unwrap();
389        z.try_extend_from_slice(&[1, 2, 3]).unwrap();
390
391        x.try_prepend(vec![1, 2, 3].try_into().unwrap())
392            .unwrap_err();
393        y.try_prepend(vec![1, 2, 3].try_into().unwrap()).unwrap();
394        z.try_prepend(vec![1, 2, 3].try_into().unwrap()).unwrap();
395
396        z.inner_mut()[0] = 0;
397
398        assert_eq!(&z.into_vec(), &[0, 2, 3, 42, 1, 2, 3]);
399        assert_eq!(TestBuffer::max_len(), N);
400    }

402    #[test]
403    fn formatting_test() {
404        use alloc::format;
405
406        let buffer = SmallTestBuffer::try_from(b"abcdefghijklmnopqrstuvwxyz012345".to_vec())
407            .expect("String is 64 bytes");
408
409        // `Debug`/`Display`.
410        assert_eq!(
411            format!("{buffer:+?}"),
412            "0x6162636465666768696a6b6c6d6e6f707172737475767778797a303132333435"
413        );
414        // `Debug`/`Display` with default precision.
415        assert_eq!(
416            format!("{buffer:?}"),
417            "0x6162636465666768..797a303132333435"
418        );
419        // `Debug`/`Display` with precision 0.
420        assert_eq!(format!("{buffer:.0?}"), "0x..");
421        // `Debug`/`Display` with precision 1.
422        assert_eq!(format!("{buffer:.1?}"), "0x61..35");
423        // `Debug`/`Display` with precision 2.
424        assert_eq!(format!("{buffer:.2?}"), "0x6162..3435");
425        // `Debug`/`Display` with precision 4.
426        assert_eq!(format!("{buffer:.4?}"), "0x61626364..32333435");
427        // `Debug`/`Display` with precision 15.
428        assert_eq!(
429            format!("{buffer:.15?}"),
430            "0x6162636465666768696a6b6c6d6e6f..72737475767778797a303132333435"
431        );
432        // `Debug`/`Display` with precision 30.
433        assert_eq!(
434            format!("{buffer:.30?}"),
435            "0x6162636465666768696a6b6c6d6e6f707172737475767778797a303132333435"
436        );
437        // Alternate formatter with default precision.
438        assert_eq!(
439            format!("{buffer:#}"),
440            "LimitedVec(0x6162636465666768..797a303132333435)"
441        );
442        // Alternate formatter with max precision.
443        assert_eq!(
444            format!("{buffer:+#}"),
445            "LimitedVec(0x6162636465666768696a6b6c6d6e6f707172737475767778797a303132333435)"
446        );
447        // Alternate formatter with precision 2.
448        assert_eq!(format!("{buffer:#.2}"), "LimitedVec(0x6162..3435)");
449    }
