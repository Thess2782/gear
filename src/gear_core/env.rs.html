36/// Lock for the payload of the incoming/currently executing message.
37///
38/// The type was mainly introduced to establish type safety mechanics
39/// for the read of the payload from externalities. To type's purposes
40/// see [`Externalities::lock_payload`] docs.
41///
42/// ### Usage
43/// This type gives access to some slice of the currently executing message
44/// payload, but doesn't do it directly. It gives to the caller the [`PayloadSliceAccess`]
45/// wrapper, which actually can return the slice of the payload. But this wrapper
46/// is instantiated only inside the [`Self::drop_with`] method.
47/// This is actually done to prevent a user of the type from locking payload of the
48/// message, which actually moves it, and forgetting to unlock it back, because
49/// if access to the slice buffer was granted directly from the holder, the type user
50/// could have written the data to memory and then have dropped the holder. As a result
51/// the executing message payload wouldn't have been returned. So [`PayloadSliceLock::drop_with`]
52/// is a kind of scope-guard for the data and the [`PayloadSliceAccess`] is a data access guard.
53///
54/// For more usage info read [`Self::drop_with`] docs.
55pub struct PayloadSliceLock {
56    /// Locked payload
57    payload: Payload,
58    /// Range values indicating slice bounds.
59    range: (usize, usize),
60}
61
62impl PayloadSliceLock {
63    /// Creates a new [`PayloadSliceLock`] from the currently executed message context.
64    ///
65    /// The method checks whether received range (slice) is correct, i.e., the end is lower
66    /// than payload's length. If the check goes well, the ownership over payload will be
67    /// taken from the message context by [`mem::take`].
68    pub fn try_new((start, end): (u32, u32), msg_ctx: &mut MessageContext) -> Option<Self> {
69        let payload_len = msg_ctx.payload_mut().inner().len();
70        if end as usize > payload_len {
71            return None;
72        }
73
74        Some(Self {
75            payload: mem::take(msg_ctx.payload_mut()),
76            range: (start as usize, end as usize),
77        })
78    }
79
80    /// Releases back ownership of the locked payload to the message context.
81    ///
82    /// The method actually performs [`mem::swap`] under the hood. It's supposed
83    /// to be called from [`Externalities::unlock_payload`], implementor of which
84    /// owns provided message context.
85    fn release(&mut self, msg_ctx: &mut MessageContext) {
86        mem::swap(msg_ctx.payload_mut(), &mut self.payload);
87    }
88
89    /// Uses the lock in the provided `job` and drops the lock after running it.
90    ///
91    /// [`PayloadSliceLock`]'s main purpose is to provide safe access to the payload's
92    /// slice and ensure it will be returned back to the message.
93    ///
94    /// Type docs explain how safe access is designed with [`PayloadSliceAccess`].
95    ///
96    /// We ensure that the payload is released back by returning the [`DropPayloadLockBound`]
97    /// from the `job`. This type can actually be instantiated only from tuple of two:
98    /// [`UnlockPayloadBound`] and some result with err variant type to be `JobErr`.
99    /// The first is returned from [`Externalities::unlock_payload`], so it means that
100    /// that payload was reclaimed by the original owner. The other result stores actual
101    /// error of the `Job` as it could have called fallible actions inside it. So,
102    /// [`DropPayloadLockBound`] gives an opportunity to store the actual result of the job,
103    /// but also gives guarantee that payload was reclaimed.
104    pub fn drop_with<JobErr, Job>(mut self, mut job: Job) -> DropPayloadLockBound<JobErr>
105    where
106        Job: FnMut(PayloadSliceAccess<'_>) -> DropPayloadLockBound<JobErr>,
107    {
108        let held_range = PayloadSliceAccess(&mut self);
109        job(held_range)
110    }
111
112    fn in_range(&self) -> &[u8] {
113        let (start, end) = self.range;
114        // Will not panic as range is checked.
115        &self.payload.inner()[start..end]
116    }
117}

119/// A wrapper over mutable reference to [`PayloadSliceLock`]
120/// which can give to the caller the slice of the held payload.
121///
122/// For more information read [`PayloadSliceLock`] docs.
123pub struct PayloadSliceAccess<'a>(&'a mut PayloadSliceLock);
124
125impl<'a> PayloadSliceAccess<'a> {
126    /// Returns slice of the held payload.
127    pub fn as_slice(&self) -> &[u8] {
128        self.0.in_range()
129    }
130
131    /// Converts the wrapper into [`PayloadSliceLock`].
132    pub fn into_lock(self) -> &'a mut PayloadSliceLock {
133        self.0
134    }
135}
136
137/// Result of calling a `job` within [`PayloadSliceLock::drop_with`].
138///
139/// This is a "bound" type which means it's main purpose is to give
140/// some type-level guarantees. More precisely, it gives guarantee
141/// that payload value was reclaimed/unlocked by the owner. Also it stores the error
142/// of the `job`, which gives opportunity to handle the actual job's runtime
143/// error, but not bound wrappers.
144pub struct DropPayloadLockBound<JobError> {
145    job_result: Result<(), JobError>,
146}
147
148impl<JobErr> DropPayloadLockBound<JobErr> {
149    /// Convert into inner job of the [`PayloadSliceLock::drop_with`] result.
150    pub fn into_inner(self) -> Result<(), JobErr> {
151        self.job_result
152    }
153}
154
155impl<JobErr> From<(UnlockPayloadBound, Result<(), JobErr>)> for DropPayloadLockBound<JobErr> {
156    fn from((_token, job_result): (UnlockPayloadBound, Result<(), JobErr>)) -> Self {
157        DropPayloadLockBound { job_result }
158    }
159}
160
161/// Result of calling [`Externalities::unlock_payload`].
162///
163/// This is a "bound" type which means it doesn't store
164/// anything, but gives type-level guarantees that [`PayloadSliceLock`]
165/// released the payload back to the message context.
166pub struct UnlockPayloadBound(());
167
168impl From<(&mut MessageContext, &mut PayloadSliceLock)> for UnlockPayloadBound {
169    fn from((msg_ctx, payload_holder): (&mut MessageContext, &mut PayloadSliceLock)) -> Self {
170        payload_holder.release(msg_ctx);
171
172        UnlockPayloadBound(())
173    }
174}

176/// External api and data for managing memory and messages,
177/// use by an executing program to trigger state transition
178/// in runtime.
179pub trait Externalities {
180    /// An error issued in infallible syscall.
181    type UnrecoverableError;
182
183    /// An error issued in fallible syscall.
184    type FallibleError;
185
186    /// An error issued during allocation.
187    type AllocError: Display;
188
189    /// Allocate number of pages.
190    ///
191    /// The resulting page number should point to `pages` consecutive memory pages.
192    fn alloc<Context>(
193        &mut self,
194        ctx: &mut Context,
195        mem: &mut impl Memory<Context>,
196        pages_num: u32,
197    ) -> Result<WasmPage, Self::AllocError>;
198
199    /// Free specific page.
200    fn free(&mut self, page: WasmPage) -> Result<(), Self::AllocError>;
201
202    /// Free specific memory range.
203    fn free_range(&mut self, start: WasmPage, end: WasmPage) -> Result<(), Self::AllocError>;
204
205    /// Get environment variables currently set in the system and in the form
206    /// corresponded to the requested version.
207    fn env_vars(&self, version: u32) -> Result<EnvVars, Self::UnrecoverableError>;
208
209    /// Get the current block height.
210    fn block_height(&self) -> Result<u32, Self::UnrecoverableError>;
211
212    /// Get the current block timestamp.
213    fn block_timestamp(&self) -> Result<u64, Self::UnrecoverableError>;
214
215    /// Initialize a new incomplete message for another program and return its handle.
216    fn send_init(&mut self) -> Result<u32, Self::FallibleError>;
217
218    /// Push an extra buffer into message payload by handle.
219    fn send_push(&mut self, handle: u32, buffer: &[u8]) -> Result<(), Self::FallibleError>;
220
221    /// Complete message and send it to another program.
222    fn send_commit(
223        &mut self,
224        handle: u32,
225        msg: HandlePacket,
226        delay: u32,
227    ) -> Result<MessageId, Self::FallibleError>;
228
229    /// Send message to another program.
230    fn send(&mut self, msg: HandlePacket, delay: u32) -> Result<MessageId, Self::FallibleError> {
231        let handle = self.send_init()?;
232        self.send_commit(handle, msg, delay)
233    }
234
235    /// Push the incoming message buffer into message payload by handle.
236    fn send_push_input(
237        &mut self,
238        handle: u32,
239        offset: u32,
240        len: u32,
241    ) -> Result<(), Self::FallibleError>;
242
243    /// Complete message and send it to another program using gas from reservation.
244    fn reservation_send_commit(
245        &mut self,
246        id: ReservationId,
247        handle: u32,
248        msg: HandlePacket,
249        delay: u32,
250    ) -> Result<MessageId, Self::FallibleError>;
251
252    /// Send message to another program using gas from reservation.
253    fn reservation_send(
254        &mut self,
255        id: ReservationId,
256        msg: HandlePacket,
257        delay: u32,
258    ) -> Result<MessageId, Self::FallibleError> {
259        let handle = self.send_init()?;
260        self.reservation_send_commit(id, handle, msg, delay)
261    }
262
263    /// Push an extra buffer into reply message.
264    fn reply_push(&mut self, buffer: &[u8]) -> Result<(), Self::FallibleError>;
265
266    /// Complete reply message and send it to source program.
267    fn reply_commit(&mut self, msg: ReplyPacket) -> Result<MessageId, Self::FallibleError>;
268
269    /// Complete reply message and send it to source program from reservation.
270    fn reservation_reply_commit(
271        &mut self,
272        id: ReservationId,
273        msg: ReplyPacket,
274    ) -> Result<MessageId, Self::FallibleError>;
275
276    /// Produce reply to the current message.
277    fn reply(&mut self, msg: ReplyPacket) -> Result<MessageId, Self::FallibleError> {
278        self.reply_commit(msg)
279    }
280
281    /// Produce reply to the current message from reservation.
282    fn reservation_reply(
283        &mut self,
284        id: ReservationId,
285        msg: ReplyPacket,
286    ) -> Result<MessageId, Self::FallibleError> {
287        self.reservation_reply_commit(id, msg)
288    }
289
290    /// Get the message id of the initial message.
291    fn reply_to(&self) -> Result<MessageId, Self::FallibleError>;
292
293    /// Get the message id which signal issues from.
294    fn signal_from(&self) -> Result<MessageId, Self::FallibleError>;
295
296    /// Push the incoming message buffer into reply message.
297    fn reply_push_input(&mut self, offset: u32, len: u32) -> Result<(), Self::FallibleError>;
298
299    /// Get the source of the message currently being handled.
300    fn source(&self) -> Result<ActorId, Self::UnrecoverableError>;
301
302    /// Get the reply code if the message being processed.
303    fn reply_code(&self) -> Result<ReplyCode, Self::FallibleError>;
304
305    /// Get the signal code if the message being processed.
306    fn signal_code(&self) -> Result<SignalCode, Self::FallibleError>;
307
308    /// Get the id of the message currently being handled.
309    fn message_id(&self) -> Result<MessageId, Self::UnrecoverableError>;
310
311    /// Get the id of program itself
312    fn program_id(&self) -> Result<ActorId, Self::UnrecoverableError>;
313
314    /// Send debug message.
315    ///
316    /// This should be no-op in release builds.
317    fn debug(&self, data: &str) -> Result<(), Self::UnrecoverableError>;
318
319    /// Takes ownership over payload of the executing message and
320    /// returns it in the wrapper [`PayloadSliceLock`], which acts
321    /// like lock.
322    ///
323    /// Due to details of implementation of the runtime which executes gear
324    /// syscalls inside wasm execution environment, to prevent additional memory
325    /// allocation on payload read op, we give ownership over payload to the caller.
326    /// Giving ownership over payload actually means, that the payload value in the
327    /// currently executed message will become empty.
328    /// To prevent from the risk of payload being not "returned" back to the
329    /// message a [`Externalities::unlock_payload`] is introduced. For more info,
330    /// read docs to [`PayloadSliceLock`], [`DropPayloadLockBound`],
331    /// [`UnlockPayloadBound`], [`PayloadSliceAccess`] types and their methods.
332    fn lock_payload(&mut self, at: u32, len: u32) -> Result<PayloadSliceLock, Self::FallibleError>;
333
334    /// Reclaims ownership from the payload lock over previously taken payload from the
335    /// currently executing message..
336    ///
337    /// It's supposed, that the implementation of the method calls `PayloadSliceLock::release`.
338    fn unlock_payload(&mut self, payload_holder: &mut PayloadSliceLock) -> UnlockPayloadBound;
339
340    /// Size of currently handled message payload.
341    fn size(&self) -> Result<usize, Self::UnrecoverableError>;
342
343    /// Returns a random seed for the current block with message id as a subject, along with the time in the past since when it was determinable by chain observers.
344    fn random(&self) -> Result<(&[u8], u32), Self::UnrecoverableError>;
345
346    /// Reserve some gas for a few blocks.
347    fn reserve_gas(
348        &mut self,
349        amount: u64,
350        duration: u32,
351    ) -> Result<ReservationId, Self::FallibleError>;
352
353    /// Unreserve gas using reservation ID.
354    fn unreserve_gas(&mut self, id: ReservationId) -> Result<u64, Self::FallibleError>;
355
356    /// Do system reservation.
357    fn system_reserve_gas(&mut self, amount: u64) -> Result<(), Self::FallibleError>;
358
359    /// Tell how much gas is left in running context.
360    fn gas_available(&self) -> Result<u64, Self::UnrecoverableError>;
361
362    /// Value associated with message.
363    fn value(&self) -> Result<u128, Self::UnrecoverableError>;
364
365    /// Tell how much value is left in running context.
366    fn value_available(&self) -> Result<u128, Self::UnrecoverableError>;
367
368    /// Interrupt the program and reschedule execution for maximum.
369    fn wait(&mut self) -> Result<(), Self::UnrecoverableError>;
370
371    /// Interrupt the program and reschedule execution in duration.
372    fn wait_for(&mut self, duration: u32) -> Result<(), Self::UnrecoverableError>;
373
374    /// Interrupt the program and reschedule execution for maximum,
375    /// but not more than duration.
376    fn wait_up_to(&mut self, duration: u32) -> Result<bool, Self::UnrecoverableError>;
377
378    /// Wake the waiting message and move it to the processing queue.
379    fn wake(&mut self, waker_id: MessageId, delay: u32) -> Result<(), Self::FallibleError>;
380
381    /// Send init message to create a new program.
382    fn create_program(
383        &mut self,
384        packet: InitPacket,
385        delay: u32,
386    ) -> Result<(MessageId, ActorId), Self::FallibleError>;
387
388    /// Create deposit to handle reply on given message.
389    fn reply_deposit(
390        &mut self,
391        message_id: MessageId,
392        amount: u64,
393    ) -> Result<(), Self::FallibleError>;
394
395    /// Return the set of functions that are forbidden to be called.
396    fn forbidden_funcs(&self) -> &BTreeSet<SyscallName>;
397
398    /// Return the current message context.
399    fn msg_ctx(&self) -> &MessageContext;
400}

417/// Trait defining type could be used as entry point for a wasm module.
418pub trait WasmEntryPoint: Sized {
419    /// Converting self into entry point name.
420    fn as_entry(&self) -> &str;
421
422    /// Converting entry point name into self object, if possible.
423    fn try_from_entry(entry: &str) -> Option<Self>;
424
425    /// Tries to convert self into `DispatchKind`.
426    fn try_into_kind(&self) -> Option<DispatchKind> {
427        <DispatchKind as WasmEntryPoint>::try_from_entry(self.as_entry())
428    }
429}
430
431impl WasmEntryPoint for String {
432    fn as_entry(&self) -> &str {
433        self
434    }
435
436    fn try_from_entry(entry: &str) -> Option<Self> {
437        Some(entry.into())
438    }
439}
440
441impl WasmEntryPoint for DispatchKind {
442    fn as_entry(&self) -> &str {
443        match self {
444            Self::Init => "init",
445            Self::Handle => "handle",
446            Self::Reply => "handle_reply",
447            Self::Signal => "handle_signal",
448        }
449    }
450
451    fn try_from_entry(entry: &str) -> Option<Self> {
452        let kind = match entry {
453            "init" => Self::Init,
454            "handle" => Self::Handle,
455            "handle_reply" => Self::Reply,
456            "handle_signal" => Self::Signal,
457            _ => return None,
458        };
459
460        Some(kind)
461    }
