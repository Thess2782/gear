60/// Entry point for dispatch processing.
61#[derive(
62    Copy, Clone, Default, Debug, Eq, Hash, Ord, PartialEq, PartialOrd, Decode, Encode, TypeInfo,
63)]
64#[cfg_attr(feature = "std", derive(serde::Serialize, serde::Deserialize))]
65pub enum DispatchKind {
66    /// Initialization.
67    Init,
68    /// Common handle.
69    #[default]
70    Handle,
71    /// Handle reply.
72    Reply,
73    /// System signal.
74    Signal,
75}
76
77impl DispatchKind {
78    /// Check if kind is init.
79    pub fn is_init(&self) -> bool {
80        matches!(self, Self::Init)
81    }
82
83    /// Check if kind is handle.
84    pub fn is_handle(&self) -> bool {
85        matches!(self, Self::Handle)
86    }
87
88    /// Check if kind is reply.
89    pub fn is_reply(&self) -> bool {
90        matches!(self, Self::Reply)
91    }
92
93    /// Check if kind is signal.
94    pub fn is_signal(&self) -> bool {
95        matches!(self, Self::Signal)
96    }
97
98    /// Returns is syscall forbidden for the dispatch kind.
99    pub fn forbids(&self, syscall_name: SyscallName) -> bool {
100        match self {
101            DispatchKind::Signal => matches!(
102                syscall_name,
103                SyscallName::Source
104                    | SyscallName::Reply
105                    | SyscallName::ReplyPush
106                    | SyscallName::ReplyCommit
107                    | SyscallName::ReplyCommitWGas
108                    | SyscallName::ReplyInput
109                    | SyscallName::ReplyInputWGas
110                    | SyscallName::ReservationReply
111                    | SyscallName::ReservationReplyCommit
112                    | SyscallName::SystemReserveGas
113            ),
114            _ => false,
115        }
116    }
117}

119/// Message packet.
120///
121/// Provides common behavior for any message's packet: accessing to payload, gas limit and value.
122pub trait Packet {
123    /// Packet payload bytes.
124    fn payload_bytes(&self) -> &[u8];
125
126    /// Payload len
127    fn payload_len(&self) -> u32;
128
129    /// Packet optional gas limit.
130    fn gas_limit(&self) -> Option<GasLimit>;
131
132    /// Packet value.
133    fn value(&self) -> Value;
134
135    /// A dispatch kind the will be generated from the packet.
136    fn kind() -> DispatchKind;
137}
