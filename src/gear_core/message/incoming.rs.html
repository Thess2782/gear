33/// Incoming message.
34///
35/// Used for program execution.
36#[derive(Clone, Default, Debug, Eq, Hash, Ord, PartialEq, PartialOrd, Decode, Encode, TypeInfo)]
37pub struct IncomingMessage {
38    /// Message id.
39    id: MessageId,
40    /// Message source.
41    source: ActorId,
42    /// Message payload.
43    payload: Payload,
44    /// Message gas limit. Required here.
45    gas_limit: GasLimit,
46    /// Message value.
47    value: Value,
48    /// Message details like reply message ID, status code, etc.
49    details: Option<MessageDetails>,
50}
51
52impl IncomingMessage {
53    /// Create new IncomingMessage.
54    pub fn new(
55        id: MessageId,
56        source: ActorId,
57        payload: Payload,
58        gas_limit: GasLimit,
59        value: Value,
60        details: Option<MessageDetails>,
61    ) -> Self {
62        Self {
63            id,
64            source,
65            payload,
66            gas_limit,
67            value,
68            details,
69        }
70    }
71
72    /// Convert IncomingMessage into gasless StoredMessage.
73    pub fn into_stored(self, destination: ActorId) -> StoredMessage {
74        StoredMessage::new(
75            self.id,
76            self.source,
77            destination,
78            self.payload,
79            self.value,
80            self.details,
81        )
82    }
83
84    /// Message id.
85    pub fn id(&self) -> MessageId {
86        self.id
87    }
88
89    /// Message source.
90    pub fn source(&self) -> ActorId {
91        self.source
92    }
93
94    /// Message payload bytes.
95    pub fn payload_bytes(&self) -> &[u8] {
96        self.payload.inner()
97    }
98
99    /// Mutable reference to message payload.
100    pub fn payload_mut(&mut self) -> &mut Payload {
101        &mut self.payload
102    }
103
104    /// Message gas limit.
105    pub fn gas_limit(&self) -> GasLimit {
106        self.gas_limit
107    }
108
109    /// Message value.
110    pub fn value(&self) -> Value {
111        self.value
112    }
113
114    /// Message details.
115    pub fn details(&self) -> Option<MessageDetails> {
116        self.details
117    }
118
119    /// Returns bool defining if message is error reply.
120    pub fn is_error_reply(&self) -> bool {
121        self.details.map(|d| d.is_error_reply()).unwrap_or(false)
122    }
123
124    /// Returns bool defining if message is reply.
125    pub fn is_reply(&self) -> bool {
126        self.details.map(|d| d.is_reply_details()).unwrap_or(false)
127    }
128}

130/// Incoming message with entry point and previous execution context, if exists.
131#[derive(Clone, Default, Debug, Eq, Hash, Ord, PartialEq, PartialOrd, Decode, Encode, TypeInfo)]
132pub struct IncomingDispatch {
133    /// Entry point.
134    kind: DispatchKind,
135    /// Incoming message.
136    message: IncomingMessage,
137    /// Previous execution context, if exists.
138    context: Option<ContextStore>,
139}
140
141impl From<IncomingDispatch> for (DispatchKind, IncomingMessage, Option<ContextStore>) {
142    fn from(dispatch: IncomingDispatch) -> (DispatchKind, IncomingMessage, Option<ContextStore>) {
143        (dispatch.kind, dispatch.message, dispatch.context)
144    }
145}
146
147impl IncomingDispatch {
148    /// Create new IncomingDispatch.
149    pub fn new(
150        kind: DispatchKind,
151        message: IncomingMessage,
152        context: Option<ContextStore>,
153    ) -> Self {
154        Self {
155            kind,
156            message,
157            context,
158        }
159    }
160
161    /// Convert IncomingDispatch into gasless StoredDispatch with updated (or recently set) context.
162    pub fn into_stored(self, destination: ActorId, context: ContextStore) -> StoredDispatch {
163        StoredDispatch::new(
164            self.kind,
165            self.message.into_stored(destination),
166            Some(context),
167        )
168    }
169
170    /// Decompose IncomingDispatch for it's components: DispatchKind, IncomingMessage and `Option<ContextStore>`.
171    pub fn into_parts(self) -> (DispatchKind, IncomingMessage, Option<ContextStore>) {
172        self.into()
173    }
174
175    /// Entry point for the message.
176    pub fn kind(&self) -> DispatchKind {
177        self.kind
178    }
179
180    /// Dispatch message reference.
181    pub fn message(&self) -> &IncomingMessage {
182        &self.message
183    }
184
185    /// Previous execution context reference, if exists.
186    pub fn context(&self) -> &Option<ContextStore> {
187        &self.context
188    }
189
190    /// Previous execution context mutable reference, if exists.
191    pub fn context_mut(&mut self) -> &mut Option<ContextStore> {
192        &mut self.context
193    }
194}
195
196impl Deref for IncomingDispatch {
197    type Target = IncomingMessage;
198
199    fn deref(&self) -> &Self::Target {
200        self.message()
201    }
