33/// Message for Reply entry point.
34/// [`ReplyMessage`] is unique because of storing [`MessageId`] from message on what it replies, and can be the only one per some message execution.
35#[derive(Clone, Default, Debug, Eq, Hash, Ord, PartialEq, PartialOrd, Decode, Encode, TypeInfo)]
36pub struct ReplyMessage {
37    /// Message id.
38    id: MessageId,
39    /// Message payload.
40    payload: Payload,
41    /// Message optional gas limit.
42    gas_limit: Option<GasLimit>,
43    /// Message value.
44    value: Value,
45    /// Reply status code.
46    code: ReplyCode,
47}
48
49impl ReplyMessage {
50    /// Create ReplyMessage from ReplyPacket.
51    pub fn from_packet(id: MessageId, packet: ReplyPacket) -> Self {
52        Self {
53            id,
54            payload: packet.payload,
55            gas_limit: packet.gas_limit,
56            value: packet.value,
57            code: packet.code,
58        }
59    }
60
61    /// Create new system generated ReplyMessage.
62    pub fn system(
63        origin_msg_id: MessageId,
64        payload: Payload,
65        value: Value,
66        err: impl Into<ErrorReplyReason>,
67    ) -> Self {
68        let id = MessageId::generate_reply(origin_msg_id);
69        let packet = ReplyPacket::system(payload, value, err);
70
71        Self::from_packet(id, packet)
72    }
73
74    /// Create new auto-generated ReplyMessage.
75    pub fn auto(origin_msg_id: MessageId) -> Self {
76        let id = MessageId::generate_reply(origin_msg_id);
77        let packet = ReplyPacket::auto();
78
79        Self::from_packet(id, packet)
80    }
81
82    /// Convert ReplyMessage into Message.
83    pub fn into_message(
84        self,
85        program_id: ActorId,
86        destination: ActorId,
87        origin_msg_id: MessageId,
88    ) -> Message {
89        Message::new(
90            self.id,
91            program_id,
92            destination,
93            self.payload,
94            self.gas_limit,
95            self.value,
96            Some(ReplyDetails::new(origin_msg_id, self.code).into()),
97        )
98    }
99
100    /// Convert ReplyMessage into StoredMessage.
101    pub fn into_stored(
102        self,
103        program_id: ActorId,
104        destination: ActorId,
105        origin_msg_id: MessageId,
106    ) -> StoredMessage {
107        self.into_message(program_id, destination, origin_msg_id)
108            .into()
109    }
110
111    /// Convert ReplyMessage into Dispatch.
112    pub fn into_dispatch(
113        self,
114        source: ActorId,
115        destination: ActorId,
116        origin_msg_id: MessageId,
117    ) -> Dispatch {
118        Dispatch::new(
119            DispatchKind::Reply,
120            self.into_message(source, destination, origin_msg_id),
121        )
122    }
123
124    /// Convert ReplyMessage into StoredDispatch.
125    pub fn into_stored_dispatch(
126        self,
127        source: ActorId,
128        destination: ActorId,
129        origin_msg_id: MessageId,
130    ) -> StoredDispatch {
131        self.into_dispatch(source, destination, origin_msg_id)
132            .into()
133    }
134
135    /// Message id.
136    pub fn id(&self) -> MessageId {
137        self.id
138    }
139
140    /// Message payload bytes.
141    pub fn payload_bytes(&self) -> &[u8] {
142        self.payload.inner()
143    }
144
145    /// Message optional gas limit.
146    pub fn gas_limit(&self) -> Option<GasLimit> {
147        self.gas_limit
148    }
149
150    /// Message value.
151    pub fn value(&self) -> Value {
152        self.value
153    }
154
155    /// Reply code of the message.
156    pub fn code(&self) -> ReplyCode {
157        self.code
158    }
159}

161/// Reply message packet.
162///
163/// This structure is preparation for future reply message sending. Has no message id.
164#[derive(Clone, Default, Debug, Eq, Hash, Ord, PartialEq, PartialOrd, Decode, Encode, TypeInfo)]
165pub struct ReplyPacket {
166    /// Message payload.
167    payload: Payload,
168    /// Message optional gas limit.
169    gas_limit: Option<GasLimit>,
170    /// Message value.
171    value: Value,
172    /// Reply status code.
173    code: ReplyCode,
174}
175
176impl ReplyPacket {
177    /// Create new manual ReplyPacket without gas.
178    pub fn new(payload: Payload, value: Value) -> Self {
179        Self {
180            payload,
181            gas_limit: None,
182            value,
183            code: ReplyCode::Success(SuccessReplyReason::Manual),
184        }
185    }
186
187    /// Create new manual ReplyPacket with gas.
188    pub fn new_with_gas(payload: Payload, gas_limit: GasLimit, value: Value) -> Self {
189        Self {
190            payload,
191            gas_limit: Some(gas_limit),
192            value,
193            code: ReplyCode::Success(SuccessReplyReason::Manual),
194        }
195    }
196
197    /// Create new manual ReplyPacket with optional gas.
198    pub fn maybe_with_gas(payload: Payload, gas_limit: Option<GasLimit>, value: Value) -> Self {
199        match gas_limit {
200            None => Self::new(payload, value),
201            Some(gas_limit) => Self::new_with_gas(payload, gas_limit, value),
202        }
203    }
204
205    // TODO: consider using here `impl CoreError` and/or provide `AsStatusCode`
206    // trait or append such functionality to `CoreError` (issue #1083).
207    /// Create new system generated ReplyPacket.
208    pub fn system(payload: Payload, value: Value, err: impl Into<ErrorReplyReason>) -> Self {
209        Self {
210            payload,
211            value,
212            code: ReplyCode::error(err),
213            ..Default::default()
214        }
215    }
216
217    /// Auto-generated reply after success execution.
218    pub fn auto() -> Self {
219        Self {
220            gas_limit: Some(0),
221            code: ReplyCode::Success(SuccessReplyReason::Auto),
222            ..Default::default()
223        }
224    }
225
226    /// Prepend payload.
227    pub(super) fn try_prepend(&mut self, mut data: Payload) -> Result<(), Payload> {
228        if data.try_extend_from_slice(self.payload_bytes()).is_err() {
229            Err(data)
230        } else {
231            self.payload = data;
232            Ok(())
233        }
234    }
235
236    /// Packet status code.
237    pub fn code(&self) -> ReplyCode {
238        self.code
239    }
240}
241
242impl Packet for ReplyPacket {
243    fn payload_bytes(&self) -> &[u8] {
244        self.payload.inner()
245    }
246
247    fn payload_len(&self) -> u32 {
248        self.payload.len_u32()
249    }
250
251    fn gas_limit(&self) -> Option<GasLimit> {
252        self.gas_limit
253    }
254
255    fn value(&self) -> Value {
256        self.value
257    }
258
259    fn kind() -> DispatchKind {
260        DispatchKind::Reply
261    }
262}
