118impl ExtManager {
119    pub(crate) fn new() -> Self {
120        Self {
121            msg_nonce: 1,
122            id_nonce: 1,
123            blocks_manager: BlocksManager::new(),
124            messages_processing_enabled: true,
125            random_data: (
126                {
127                    let mut rng = StdRng::seed_from_u64(INITIAL_RANDOM_SEED);
128                    let mut random = [0u8; 32];
129                    rng.fill_bytes(&mut random);
130
131                    random.to_vec()
132                },
133                0,
134            ),
135            ..Default::default()
136        }
137    }
138
139    pub fn block_height(&self) -> u32 {
140        self.blocks_manager.get().height
141    }
142
143    pub(crate) fn store_new_actor(
144        &mut self,
145        program_id: ActorId,
146        program: Program,
147        init_message_id: Option<MessageId>,
148    ) -> Option<TestActor> {
149        Actors::insert(program_id, TestActor::new(init_message_id, program))
150    }
151
152    pub(crate) fn store_new_code(&mut self, code_id: CodeId, code: Vec<u8>) {
153        self.opt_binaries.insert(code_id, code);
154    }
155
156    pub(crate) fn read_code(&self, code_id: CodeId) -> Option<&[u8]> {
157        self.opt_binaries.get(&code_id).map(|code| code.as_ref())
158    }
159
160    pub(crate) fn fetch_inc_message_nonce(&mut self) -> u64 {
161        let nonce = self.msg_nonce;
162        self.msg_nonce += 1;
163        nonce
164    }
165
166    pub(crate) fn free_id_nonce(&mut self) -> u64 {
167        while Actors::contains_key(self.id_nonce.into()) {
168            self.id_nonce += 1;
169        }
170        self.id_nonce
171    }
172
173    /// Check if the current block number should trigger new epoch and reset
174    /// the provided random data.
175    pub(crate) fn check_epoch(&mut self) {
176        let block_height = self.block_height();
177        if block_height % EPOCH_DURATION_IN_BLOCKS == 0 {
178            let mut rng = StdRng::seed_from_u64(
179                INITIAL_RANDOM_SEED + (block_height / EPOCH_DURATION_IN_BLOCKS) as u64,
180            );
181            let mut random = [0u8; 32];
182            rng.fill_bytes(&mut random);
183
184            self.random_data = (random.to_vec(), block_height + 1);
185        }
186    }
187
188    pub(crate) fn update_storage_pages(
189        &mut self,
190        program_id: &ActorId,
191        memory_pages: BTreeMap<GearPage, PageBuf>,
192    ) {
193        Actors::modify(*program_id, |actor| {
194            let pages_data = actor
195                .unwrap_or_else(|| panic!("Actor id {program_id:?} not found"))
196                .get_pages_data_mut()
197                .expect("No pages data found for program");
198
199            for (page, buf) in memory_pages {
200                pages_data.insert(page, buf);
201            }
202        });
203    }
204
205    pub(crate) fn mint_to(&mut self, id: &ActorId, value: Value) {
206        Accounts::increase(*id, value);
207    }
208
209    pub(crate) fn balance_of(&self, id: &ActorId) -> Value {
210        Accounts::balance(*id)
211    }
212
213    pub(crate) fn override_balance(&mut self, &id: &ActorId, balance: Value) {
214        if Actors::is_user(id) && balance < crate::EXISTENTIAL_DEPOSIT {
215            usage_panic!(
216                "An attempt to override balance with value ({}) less than existential deposit ({}. \
217                Please try to use bigger balance value",
218                balance,
219                crate::EXISTENTIAL_DEPOSIT
220            );
221        }
222        Accounts::override_balance(id, balance);
223    }
224
225    pub(crate) fn on_task_pool_change(&mut self) {
226        let write = DbWeights::default().write.ref_time;
227        self.gas_allowance = self.gas_allowance.saturating_sub(Gas(write));
228    }
229
230    fn init_success(&mut self, program_id: ActorId) {
231        Actors::modify(program_id, |actor| {
232            actor
233                .unwrap_or_else(|| panic!("Actor id {program_id:?} not found"))
234                .set_initialized()
235        });
236    }
237
238    fn init_failure(&mut self, program_id: ActorId, origin: ActorId) {
239        Actors::modify(program_id, |actor| {
240            let actor = actor.unwrap_or_else(|| panic!("Actor id {program_id:?} not found"));
241            *actor = TestActor::FailedInit;
242        });
243
244        let value = Accounts::balance(program_id);
245        if value != 0 {
246            Accounts::transfer(program_id, origin, value, false);
247        }
248    }
249
250    pub(crate) fn update_genuine_program<R, F: FnOnce(&mut GenuineProgram) -> R>(
251        &mut self,
252        id: ActorId,
253        op: F,
254    ) -> Option<R> {
255        Actors::modify(id, |actor| {
256            actor.and_then(|actor| actor.genuine_program_mut().map(op))
257        })
258    }
259
260    pub(crate) fn read_mailbox_message(
261        &mut self,
262        to: ActorId,
263        from_mid: MessageId,
264    ) -> Result<UserStoredMessage, MailboxErrorImpl> {
265        let (message, hold_interval) = self.mailbox.remove(to, from_mid)?;
266
267        let expected = hold_interval.finish;
268
269        let user_id = message.destination();
270        let from = message.source();
271
272        self.charge_for_hold(message.id(), hold_interval, StorageType::Mailbox);
273        self.consume_and_retrieve(message.id());
274
275        self.bank.transfer_value(from, user_id, message.value());
276
277        let _ = self
278            .task_pool
279            .delete(
280                expected,
281                ScheduledTask::RemoveFromMailbox(user_id, message.id()),
282            )
283            .map(|_| {
284                self.on_task_pool_change();
285            });
286
287        Ok(message)
288    }
