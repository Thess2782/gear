19use super::*;

21impl ExtManager {
22    /// Call non-void meta function from actor stored in manager.
23    /// Warning! This is a static call that doesn't change actors pages data.
24    pub(crate) fn read_state_bytes(
25        &mut self,
26        payload: Vec<u8>,
27        program_id: &ActorId,
28    ) -> Result<Vec<u8>> {
29        let executable_actor_data = Actors::modify(*program_id, |actor| {
30            if let Some(actor) = actor {
31                Ok(actor.get_executable_actor_data())
32            } else {
33                Err(TestError::ActorNotFound(*program_id))
34            }
35        })?;
36
37        if let Some((data, code)) = executable_actor_data {
38            core_processor::informational::execute_for_reply::<Ext<LazyPagesNative>, _>(
39                String::from("state"),
40                code,
41                Some(data.allocations),
42                Some((*program_id, Default::default())),
43                payload,
44                MAX_USER_GAS_LIMIT,
45                self.blocks_manager.get(),
46            )
47            .map_err(TestError::ReadStateError)
48        } else if let Some(mut program_mock) = Actors::modify(*program_id, |actor| {
49            actor.expect("Checked before").take_mock()
50        }) {
51            program_mock
52                .state()
53                .map_err(|err| TestError::ReadStateError(err.into()))
54        } else {
55            Err(TestError::ActorIsNotExecutable(*program_id))
56        }
57    }
58    pub(crate) fn read_memory_pages(&self, program_id: &ActorId) -> BTreeMap<GearPage, PageBuf> {
59        Actors::access(*program_id, |actor| {
60            let program = match actor.unwrap_or_else(|| panic!("Actor id {program_id:?} not found"))
61            {
62                TestActor::Initialized(program) => program,
63                TestActor::Uninitialized(_, program) => program.as_ref().unwrap(),
64                TestActor::Exited(_) => panic!("Actor {program_id} is exited"),
65                TestActor::FailedInit => panic!("Actor {program_id} failed to init"),
66                TestActor::CodeNotExists => panic!("Actor {program_id} code not exists"),
67            };
68
69            match program {
70                Program::Genuine(program) => program.pages_data.clone(),
71                Program::Mock(_) => panic!("Can't read memory of mock program"),
72            }
73        })
74    }
75}
