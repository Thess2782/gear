46#[derive(Decode)]
47struct PageKey {
48    _page_storage_prefix: [u8; 32],
49    program_id: ActorId,
50    _memory_infix: u32,
51    page: GearPage,
52}

54#[derive(Debug)]
55struct PagesStorage;

57impl LazyPagesStorage for PagesStorage {
58    fn page_exists(&self, mut key: &[u8]) -> bool {
59        let PageKey {
60            program_id, page, ..
61        } = PageKey::decode_all(&mut key).expect("Invalid key");
62
63        Actors::access(program_id, |actor| {
64            actor
65                .and_then(|actor| actor.get_pages_data())
66                .map(|pages_data| pages_data.contains_key(&page))
67                .unwrap_or(false)
68        })
69    }
70
71    fn load_page(&mut self, mut key: &[u8], buffer: &mut [u8]) -> Option<u32> {
72        let PageKey {
73            program_id, page, ..
74        } = PageKey::decode_all(&mut key).expect("Invalid key");
75
76        Actors::access(program_id, |actor| {
77            actor
78                .and_then(|actor| actor.get_pages_data())
79                .and_then(|pages_data| pages_data.get(&page))
80                .map(|page_buf| {
81                    buffer.copy_from_slice(page_buf);
82                    page_buf.len() as u32
83                })
84        })
85    }
86}

88/// The testing environment which simulates the chain state and its
89/// transactions but somehow the real on-chain execution environment
90/// could be different.
91///
92/// ```
93/// use gtest::System;
94///
95/// // Create a new testing environment.
96/// let system = System::new();
97///
98/// // Init logger with "gwasm" target set to `debug` level.
99/// system.init_logger();
100/// ```
101pub struct System(pub(crate) RefCell<ExtManager>);

107    /// Create a new testing environment.
108    ///
109    /// # Panics
110    /// Only one instance in the current thread of the `System` is possible to
111    /// create. Instantiation of the other one leads to runtime panic.
112    #[allow(clippy::new_without_default)]
113    pub fn new() -> Self {
114        SYSTEM_INITIALIZED.with_borrow_mut(|initialized| {
115            if *initialized {
116                panic!("Impossible to have multiple instances of the `System`.");
117            }
118
119            let ext_manager = ExtManager::new();
120            gear_lazy_pages::init(
121                LazyPagesVersion::Version1,
122                LazyPagesInitContext::new(Self::PAGE_STORAGE_PREFIX),
123                PagesStorage,
124            )
125            .expect("Failed to init lazy-pages");
126
127            *initialized = true;
128
129            Self(RefCell::new(ext_manager))
130        })
131    }

157    /// Run next block.
158    ///
159    /// Block execution model is the following:
160    /// - increase the block number, update the timestamp
161    /// - process tasks from the task pool
162    /// - process messages in the queue.
163    ///
164    /// The system is always initialized with a 0 block number. Current block
165    /// number in the system is the number of the already executed block,
166    /// therefore block execution starts with a block info update (block
167    /// number increase, timestamp update). For example, if current block
168    /// number is 2, it means that messages and tasks on 2 were executed, so
169    /// the method goes to block number 3 and executes tasks and messages for
170    /// the updated block number.
171    ///
172    /// Task processing basically tries to execute the scheduled to the specific
173    /// block tasks:
174    /// - delayed sending
175    /// - waking message
176    /// - removing from the mailbox
177    /// - removing reservations
178    /// - removing stalled wait message.
179    ///
180    /// Messages processing executes messages until either queue becomes empty
181    /// or block gas allowance is fully consumed.
182    pub fn run_next_block(&self) -> BlockRunResult {
183        self.run_next_block_with_allowance(Gas(GAS_ALLOWANCE))
184    }
185
186    /// Runs blocks same as [`Self::run_next_block`], but with limited
187    /// allowance.
188    pub fn run_next_block_with_allowance(&self, allowance: Gas) -> BlockRunResult {
189        if allowance > Gas(GAS_ALLOWANCE) {
190            usage_panic!(
191                "Provided allowance more than allowed limit of {GAS_ALLOWANCE}. \
192                Please, provide an allowance less than or equal to the limit."
193            );
194        }
195
196        self.0.borrow_mut().run_new_block(allowance)
197    }
198
199    /// Runs blocks same as [`Self::run_next_block`], but executes blocks to
200    /// block number `bn` including it.
201    pub fn run_to_block(&self, bn: u32) -> Vec<BlockRunResult> {
202        let mut manager = self.0.borrow_mut();
203
204        let mut current_block = manager.block_height();
205        if current_block > bn {
206            usage_panic!("Can't run blocks until bn {bn}, as current bn is {current_block}");
207        }
208
209        let mut ret = Vec::with_capacity((bn - current_block) as usize);
210        while current_block != bn {
211            let res = manager.run_new_block(Gas(GAS_ALLOWANCE));
212            ret.push(res);
213
214            current_block = manager.block_height();
215        }
216
217        ret
218    }
219
220    /// Runs `amount` of blocks only with processing task pool, without
221    /// processing the message queue.
222    pub fn run_scheduled_tasks(&self, amount: u32) -> Vec<BlockRunResult> {
223        let mut manager = self.0.borrow_mut();
224        let block_height = manager.block_height();
225
226        (block_height..block_height + amount)
227            .map(|_| {
228                manager.check_epoch();
229
230                let block_info = manager.blocks_manager.next_block();
231                let next_block_number = block_info.height;
232                manager.process_tasks(next_block_number);
233
234                let log = mem::take(&mut manager.log)
235                    .into_iter()
236                    .map(CoreLog::from)
237                    .collect();
238                BlockRunResult {
239                    block_info,
240                    gas_allowance_spent: Gas(GAS_ALLOWANCE) - manager.gas_allowance,
241                    log,
242                    ..Default::default()
243                }
244            })
245            .collect()
246    }

338    /// Saves code to the storage and returns its code hash
339    ///
340    /// This method is mainly used for providing a proper program from program
341    /// creation logic. In order to successfully create a new program with
342    /// `gstd::prog::create_program_bytes_with_gas` function, developer should
343    /// provide to the function "child's" code hash. Code for that code hash
344    /// must be in storage at the time of the function call. So this method
345    /// stores the code in storage.
346    pub fn submit_code(&self, binary: impl Into<Vec<u8>>) -> CodeId {
347        let code = binary.into();
348        let code_id = CodeId::generate(code.as_ref());
349        self.0.borrow_mut().store_new_code(code_id, code);
350
351        code_id
352    }

412impl Drop for System {
413    fn drop(&mut self) {
414        // Uninitialize
415        SYSTEM_INITIALIZED.with_borrow_mut(|initialized| *initialized = false);
416        self.0.borrow().gas_tree.reset();
417        self.0.borrow().mailbox.reset();
418        self.0.borrow().task_pool.clear();
419        self.0.borrow().waitlist.reset();
420
421        // Clear actors and accounts storages
422        Actors::clear();
423        Accounts::clear();
424    }
425}

456    #[test]
457    fn test_bn_adjustments() {
458        let sys = System::new();
459        assert_eq!(sys.block_height(), 0);
460
461        // ### Check block info after run to next block ###
462        let res = sys.run_next_block();
463        let block_info = res.block_info;
464        assert_eq!(block_info.height, sys.block_height());
465        assert_eq!(block_info.height, 1);
466
467        // ### Check block info after run to block ###
468        let current_height = block_info.height;
469        let until_height = 5;
470        let results = sys.run_to_block(until_height);
471        assert_eq!(results.len(), (until_height - current_height) as usize);
472
473        // Check first block executed is always the next block
474        let first_run = results.first().expect("checked above");
475        assert_eq!(first_run.block_info.height, current_height + 1);
476
477        // Check the last block executed number
478        let last_run = results.last().expect("checked above");
479        assert_eq!(last_run.block_info.height, until_height);
480        assert_eq!(last_run.block_info.height, sys.block_height());
481
482        // ### Check block info after running the task pool ###
483        let current_height = last_run.block_info.height;
484        let amount_of_blocks = 10;
485        let results = sys.run_scheduled_tasks(amount_of_blocks);
486        assert_eq!(results.len(), amount_of_blocks as usize);
487
488        let first_run = results.first().expect("checked above");
489        assert_eq!(first_run.block_info.height, current_height + 1);
490
491        let last_run = results.last().expect("checked above");
492        assert_eq!(
493            last_run.block_info.height,
494            current_height + amount_of_blocks
495        );
496
497        assert_eq!(last_run.block_info.height, 15);
498    }
