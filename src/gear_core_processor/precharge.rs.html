37/// Operation related to gas charging.
38#[derive(Debug, PartialEq, Eq, derive_more::Display)]
39pub enum PreChargeGasOperation {
40    /// Handle memory static pages.
41    #[display("handle memory static pages")]
42    StaticPages,
43    /// Handle program data.
44    #[display("handle program data")]
45    ProgramData,
46    /// Obtain code length.
47    #[display("obtain program code length")]
48    ProgramCodeLen,
49    /// Handle program code.
50    #[display("handle program code")]
51    ProgramCode,
52    /// Instantiate the type section of the Wasm module.
53    #[display("instantiate {_0} of Wasm module")]
54    ModuleInstantiation(SectionName),
55    /// Instrument Wasm module.
56    #[display("instrument Wasm module")]
57    ModuleInstrumentation,
58    /// Obtain program allocations.
59    #[display("obtain program allocations")]
60    Allocations,
61}
62
63#[derive(Debug, Eq, PartialEq)]
64enum PrechargeError {
65    BlockGasExceeded,
66    GasExceeded(PreChargeGasOperation),
67}
68
69struct GasPrecharger<'a> {
70    counter: &'a mut GasCounter,
71    allowance_counter: &'a mut GasAllowanceCounter,
72    costs: &'a ProcessCosts,
73}
74
75impl<'a> GasPrecharger<'a> {
76    pub fn new(
77        counter: &'a mut GasCounter,
78        allowance_counter: &'a mut GasAllowanceCounter,
79        costs: &'a ProcessCosts,
80    ) -> Self {
81        Self {
82            counter,
83            allowance_counter,
84            costs,
85        }
86    }
87
88    fn charge_gas(
89        &mut self,
90        operation: PreChargeGasOperation,
91        amount: u64,
92    ) -> Result<(), PrechargeError> {
93        if self.allowance_counter.charge_if_enough(amount) != ChargeResult::Enough {
94            return Err(PrechargeError::BlockGasExceeded);
95        }
96        if self.counter.charge_if_enough(amount) != ChargeResult::Enough {
97            return Err(PrechargeError::GasExceeded(operation));
98        }
99
100        Ok(())
101    }
102
103    pub fn charge_gas_for_program_data(&mut self) -> Result<(), PrechargeError> {
104        self.charge_gas(
105            PreChargeGasOperation::ProgramData,
106            self.costs.read.cost_for_one(),
107        )
108    }
109
110    pub fn charge_gas_for_program_code_len(&mut self) -> Result<(), PrechargeError> {
111        self.charge_gas(
112            PreChargeGasOperation::ProgramCodeLen,
113            self.costs.read.cost_for_one(),
114        )
115    }
116
117    pub fn charge_gas_for_program_code(
118        &mut self,
119        code_len: BytesAmount,
120    ) -> Result<(), PrechargeError> {
121        self.charge_gas(
122            PreChargeGasOperation::ProgramCode,
123            self.costs
124                .read
125                .cost_for_with_bytes(self.costs.read_per_byte, code_len),
126        )
127    }
128
129    pub fn charge_gas_for_section_instantiation(
130        &mut self,
131        section_name: SectionName,
132        section_len: BytesAmount,
133    ) -> Result<(), PrechargeError> {
134        let instantiation_costs = &self.costs.instantiation_costs;
135
136        let cost_per_byte = match section_name {
137            SectionName::Function => &instantiation_costs.code_section_per_byte,
138            SectionName::Data => &instantiation_costs.data_section_per_byte,
139            SectionName::Global => &instantiation_costs.global_section_per_byte,
140            SectionName::Table => &instantiation_costs.table_section_per_byte,
141            SectionName::Element => &instantiation_costs.element_section_per_byte,
142            SectionName::Type => &instantiation_costs.type_section_per_byte,
143            _ => {
144                // TODO: change this to a system error in future
145                unimplemented!("Wrong {section_name:?} for section instantiation")
146            }
147        };
148
149        self.charge_gas(
150            PreChargeGasOperation::ModuleInstantiation(section_name),
151            cost_per_byte.cost_for(section_len),
152        )
153    }
154
155    pub fn charge_gas_for_instrumentation(
156        &mut self,
157        original_code_len_bytes: BytesAmount,
158    ) -> Result<(), PrechargeError> {
159        self.charge_gas(
160            PreChargeGasOperation::ModuleInstrumentation,
161            self.costs
162                .instrumentation
163                .cost_for_with_bytes(self.costs.instrumentation_per_byte, original_code_len_bytes),
164        )
165    }
166}

168/// Possible variants of the [`DispatchResult`] if the latter contains value.
169#[allow(missing_docs)]
170#[derive(Debug)]
171pub enum SuccessfulDispatchResultKind {
172    Exit(ActorId),
173    Wait(Option<u32>, MessageWaitedType),
174    Success,
175}
176
177/// Defines result variants of the precharge functions.
178pub type PrechargeResult<T> = Result<T, Vec<JournalNote>>;
179
180/// Charge a message for program data beforehand.
181pub fn precharge_for_program(
182    block_config: &BlockConfig,
183    gas_allowance: u64,
184    dispatch: IncomingDispatch,
185    destination_id: ActorId,
186) -> PrechargeResult<ContextChargedForProgram> {
187    let mut gas_counter = GasCounter::new(dispatch.gas_limit());
188    let mut gas_allowance_counter = GasAllowanceCounter::new(gas_allowance);
189    let mut charger = GasPrecharger::new(
190        &mut gas_counter,
191        &mut gas_allowance_counter,
192        &block_config.costs,
193    );
194
195    match charger.charge_gas_for_program_data() {
196        Ok(()) => Ok(ContextChargedForProgram {
197            dispatch,
198            destination_id,
199            gas_counter,
200            gas_allowance_counter,
201        }),
202        Err(PrechargeError::BlockGasExceeded) => {
203            let gas_burned = gas_counter.burned();
204            Err(process_allowance_exceed(
205                dispatch,
206                destination_id,
207                gas_burned,
208            ))
209        }
210        Err(PrechargeError::GasExceeded(op)) => {
211            let gas_burned = gas_counter.burned();
212            let system_reservation_ctx = SystemReservationContext::from_dispatch(&dispatch);
213            Err(process_execution_error(
214                dispatch,
215                destination_id,
216                gas_burned,
217                system_reservation_ctx,
218                ActorExecutionErrorReplyReason::PreChargeGasLimitExceeded(op),
219            ))
220        }
221    }
222}

224/// Precharge for allocations obtaining from storage.
225pub fn precharge_for_allocations(
226    block_config: &BlockConfig,
227    mut context: ContextChargedForProgram,
228    allocations_tree_len: u32,
229) -> PrechargeResult<ContextChargedForAllocations> {
230    let mut charger = GasPrecharger::new(
231        &mut context.gas_counter,
232        &mut context.gas_allowance_counter,
233        &block_config.costs,
234    );
235
236    if allocations_tree_len == 0 {
237        return Ok(ContextChargedForAllocations(context));
238    }
239
240    let amount = block_config
241        .costs
242        .load_allocations_per_interval
243        .cost_for(allocations_tree_len)
244        .saturating_add(block_config.costs.read.cost_for_one());
245
246    match charger.charge_gas(PreChargeGasOperation::Allocations, amount) {
247        Ok(()) => Ok(ContextChargedForAllocations(context)),
248        Err(PrechargeError::BlockGasExceeded) => {
249            let gas_burned = context.gas_counter.burned();
250            Err(process_allowance_exceed(
251                context.dispatch,
252                context.destination_id,
253                gas_burned,
254            ))
255        }
256        Err(PrechargeError::GasExceeded(op)) => {
257            let gas_burned = context.gas_counter.burned();
258            let system_reservation_ctx = SystemReservationContext::from_dispatch(&context.dispatch);
259            Err(process_execution_error(
260                context.dispatch,
261                context.destination_id,
262                gas_burned,
263                system_reservation_ctx,
264                ActorExecutionErrorReplyReason::PreChargeGasLimitExceeded(op),
265            ))
266        }
267    }
268}

270/// Charge a message for fetching the actual length of the binary code
271/// from a storage. The updated value of binary code length
272/// should be kept in standalone storage. The caller has to call this
273/// function to charge gas-counters accordingly before fetching the value.
274///
275/// The function also performs several additional checks:
276/// - if an actor is executable
277/// - if a required dispatch method is exported.
278pub fn precharge_for_code_length(
279    block_config: &BlockConfig,
280    context: ContextChargedForAllocations,
281    actor_data: ExecutableActorData,
282) -> PrechargeResult<ContextChargedForCodeLength> {
283    let ContextChargedForProgram {
284        dispatch,
285        destination_id,
286        mut gas_counter,
287        mut gas_allowance_counter,
288    } = context.0;
289
290    if !actor_data.code_exports.contains(&dispatch.kind()) {
291        return Err(process_success(
292            SuccessfulDispatchResultKind::Success,
293            DispatchResult::success(dispatch, destination_id, gas_counter.to_amount()),
294        ));
295    }
296
297    let mut charger = GasPrecharger::new(
298        &mut gas_counter,
299        &mut gas_allowance_counter,
300        &block_config.costs,
301    );
302    match charger.charge_gas_for_program_code_len() {
303        Ok(()) => Ok(ContextChargedForCodeLength {
304            data: ContextData {
305                gas_counter,
306                gas_allowance_counter,
307                dispatch,
308                destination_id,
309                actor_data,
310            },
311        }),
312        Err(PrechargeError::BlockGasExceeded) => Err(process_allowance_exceed(
313            dispatch,
314            destination_id,
315            gas_counter.burned(),
316        )),
317        Err(PrechargeError::GasExceeded(op)) => {
318            let system_reservation_ctx = SystemReservationContext::from_dispatch(&dispatch);
319            Err(process_execution_error(
320                dispatch,
321                destination_id,
322                gas_counter.burned(),
323                system_reservation_ctx,
324                ActorExecutionErrorReplyReason::PreChargeGasLimitExceeded(op),
325            ))
326        }
327    }
328}

331pub fn precharge_for_code(
332    block_config: &BlockConfig,

351                SystemReservationContext::from_dispatch(&context.data.dispatch);

354                context.data.destination_id,
355                context.data.gas_counter.burned(),
356                system_reservation_ctx,
357                ActorExecutionErrorReplyReason::PreChargeGasLimitExceeded(op),
358            ))
359        }
360    }

396/// Charge a message for program memory and module instantiation beforehand.
397pub fn precharge_for_module_instantiation(
398    block_config: &BlockConfig,
399    mut context: ContextChargedForInstrumentation,
400    section_sizes: &InstantiatedSectionSizes,
401) -> PrechargeResult<ContextChargedForMemory> {
402    let ContextChargedForInstrumentation {
403        data:
404            ContextData {
405                gas_counter,
406                gas_allowance_counter,
407                actor_data,
408                ..
409            },
410        ..
411    } = &mut context;
412
413    let mut f = || {
414        let mut charger =
415            GasPrecharger::new(gas_counter, gas_allowance_counter, &block_config.costs);
416
417        // Calculates size of wasm memory buffer which must be created in execution environment
418        let memory_size = if let Some(page) = actor_data.allocations.end() {
419            page.inc()
420        } else {
421            actor_data.static_pages
422        };
423
424        charger.charge_gas_for_section_instantiation(
425            SectionName::Function,
426            section_sizes.code_section.into(),
427        )?;
428        charger.charge_gas_for_section_instantiation(
429            SectionName::Data,
430            section_sizes.data_section.into(),
431        )?;
432        charger.charge_gas_for_section_instantiation(
433            SectionName::Global,
434            section_sizes.global_section.into(),
435        )?;
436        charger.charge_gas_for_section_instantiation(
437            SectionName::Table,
438            section_sizes.table_section.into(),
439        )?;
440        charger.charge_gas_for_section_instantiation(
441            SectionName::Element,
442            section_sizes.element_section.into(),
443        )?;
444        charger.charge_gas_for_section_instantiation(
445            SectionName::Type,
446            section_sizes.type_section.into(),
447        )?;
448
449        Ok(memory_size)
450    };
451
452    match f() {
453        Ok(memory_size) => {
454            log::trace!("Charged for module instantiation and memory pages. Size: {memory_size:?}");
455            Ok(ContextChargedForMemory {
456                data: context.data,
457                max_reservations: block_config.max_reservations,
458                memory_size,
459            })
460        }
461        Err(err) => {
462            log::trace!("Failed to charge for module instantiation or memory pages: {err:?}");
463            match err {
464                PrechargeError::BlockGasExceeded => Err(process_allowance_exceed(
465                    context.data.dispatch,
466                    context.data.destination_id,
467                    context.data.gas_counter.burned(),
468                )),
469                PrechargeError::GasExceeded(op) => {
470                    let system_reservation_ctx =
471                        SystemReservationContext::from_dispatch(&context.data.dispatch);
472                    Err(process_execution_error(
473                        context.data.dispatch,
474                        context.data.destination_id,
475                        context.data.gas_counter.burned(),
476                        system_reservation_ctx,
477                        ActorExecutionErrorReplyReason::PreChargeGasLimitExceeded(op),
478                    ))
479                }
480            }
481        }
482    }
